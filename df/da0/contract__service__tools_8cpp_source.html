84nlohmann::json ReadJSConfig(const std::string& config_path) {
85
86 std::ifstream contract_fstream(config_path);

103int main(int argc, char** argv) {
104 if (argc < 2) {
105 printf("<cmd> -c [config]\n");
106 return 0;
107 }
108
109
110 std::string config_file;
111
112 std::string cmd;
113 std::string caller_address, contract_name, contract_path, params,
114 contract_address, func_name;
115 int c;
116 int option_index;
117 std::string client_config_file;
118 while ((c = getopt_long(argc, argv, "c:h", long_options, &option_index)) != -1) {
119 switch (c) {
120 case -1:
121 break;
122 case 'f':
123 config_file = optarg;
124 break;
125 case 'c':
126 client_config_file = optarg;
127 break;
128 case 'h':
129 ShowUsage();
130 break;
131 }
132 }
133
134 nlohmann::json js = ReadJSConfig(config_file);
135 std::cout<<js<<std::endl;
136 cmd = GetValue(js, "command");
137
138 ResDBConfig config = GenerateResDBConfig(client_config_file);
139 config.SetClientTimeoutMs(100000);
140
141 ContractClient client(config);
142 printf("cmd = %s\n", cmd.c_str());
143 if (cmd == "create_account") {
144 CreateAccount(&client);
145 } else if (cmd == "deploy") {
146
147 contract_path = GetValue(js, "contract_path");
148 contract_name = GetValue(js, "contract_name");
149 contract_address = GetValue(js, "owner_address");
150 params = GetValue(js, "init_params");
151
152 printf("contract path %s cmd %s contract name %s caller_address %s init params %s\n", contract_path.c_str(), cmd.c_str(), contract_name.c_str(), contract_address.c_str(), params.c_str());
153
154 std::vector<std::string> init_params;
155 boost::split(init_params, params, boost::is_any_of(","));
156
157 DeployContract(&client, contract_address, contract_name, contract_path,
158 init_params);
159 } else if (cmd == "execute") {
160
161 caller_address = GetValue(js, "caller_address");
162 contract_address = GetValue(js, "contract_address");
163 func_name = GetValue(js, "func_name");
164 params = GetValue(js, "params");
165
166 printf(
167 "execute\n caller address:%s\n contract address: %s\n func: %s\n "
168 "params:%s\n",
169 caller_address.c_str(), contract_address.c_str(), func_name.c_str(),
170 params.c_str());
171 std::vector<std::string> func_params;
172 boost::split(func_params, params, boost::is_any_of(","));
173
174 ExecuteContract(&client, caller_address, contract_address, func_name,
175 func_params);
176 }
177 else if (cmd == "get_balance") {
178 std::string address = GetValue(js, "address");
179 auto balance_or = client.GetBalance(address);
180 printf("get address %s balance %s\n", address.c_str(), (*balance_or).c_str());
181 } else if (cmd == "set_balance") {
182 std::string address = GetValue(js, "address");
183 std::string balance = GetValue(js, "balance");
184 printf("address %s balance %s\n", address.c_str(), balance.c_str());
185 auto ret = client.SetBalance(address, balance);
186 printf("set address %s balance %s ret %s\n", address.c_str(), balance.c_str(), (*ret).c_str());
187 }
188
189}
