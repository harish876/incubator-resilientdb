64class Input(object):
65 """A Input is used to spend assets locked by an Output.
66
67 Wraps around a Crypto-condition Fulfillment.
68
69 Attributes:
70 fulfillment (:class:`cryptoconditions.Fulfillment`): A Fulfillment
71 to be signed with a private key.
72 owners_before (:obj:`list` of :obj:`str`): A list of owners after a
73 Transaction was confirmed.
74 fulfills (:class:`~resdb_validator.transaction. TransactionLink`,
75 optional): A link representing the input of a `TRANSFER`
76 Transaction.
77 """
78
79 def __init__(self, fulfillment, owners_before, fulfills=None):
80 """Create an instance of an :class:`~.Input`.
81
82 Args:
83 fulfillment (:class:`cryptoconditions.Fulfillment`): A
84 Fulfillment to be signed with a private key.
85 owners_before (:obj:`list` of :obj:`str`): A list of owners
86 after a Transaction was confirmed.
87 fulfills (:class:`~resdb_validator.transaction.
88 TransactionLink`, optional): A link representing the input
89 of a `TRANSFER` Transaction.
90 """
91 if fulfills is not None and not isinstance(fulfills, TransactionLink):
92 raise TypeError('`fulfills` must be a TransactionLink instance')
93 if not isinstance(owners_before, list):
94 raise TypeError('`owners_before` must be a list instance')
95
96 self.fulfillment = fulfillment
97 self.fulfills = fulfills
98 self.owners_before = owners_before
99
100 def __eq__(self, other):
101 # TODO: If `other !== Fulfillment` return `False`
102 return self.to_dict() == other.to_dict()
103
104 # NOTE: This function is used to provide a unique key for a given
105 # Input to suppliment memoization
106 def __hash__(self):
107 return hash((self.fulfillment, self.fulfills))
108
109 def to_dict(self):
110 """Transforms the object to a Python dictionary.
111
112 Note:
113 If an Input hasn't been signed yet, this method returns a
114 dictionary representation.
115
116 Returns:
117 dict: The Input as an alternative serialization format.
118 """
119 try:
120 fulfillment = self.fulfillment.serialize_uri()
121 except (TypeError, AttributeError, ASN1EncodeError, ASN1DecodeError):
122 fulfillment = _fulfillment_to_details(self.fulfillment)
123
124 try:
125 # NOTE: `self.fulfills` can be `None` and that's fine
126 fulfills = self.fulfills.to_dict()
127 except AttributeError:
128 fulfills = None
129
130 input_ = {
131 'owners_before': self.owners_before,
132 'fulfills': fulfills,
133 'fulfillment': fulfillment,
134 }
135 return input_

178def _fulfillment_to_details(fulfillment):
179 """Encode a fulfillment as a details dictionary
180
181 Args:
182 fulfillment: Crypto-conditions Fulfillment object
183 """
184
185 if fulfillment.type_name == 'ed25519-sha-256':
186 return {
187 'type': 'ed25519-sha-256',
188 'public_key': base58.b58encode(fulfillment.public_key).decode(),
189 }
190
191 if fulfillment.type_name == 'threshold-sha-256':
192 subconditions = [
193 _fulfillment_to_details(cond['body'])
194 for cond in fulfillment.subconditions
195 ]
196 return {
197 'type': 'threshold-sha-256',
198 'threshold': fulfillment.threshold,
199 'subconditions': subconditions,
200 }
201
202 raise UnsupportedTypeError(fulfillment.type_name)
203
204
205def _fulfillment_from_details(data, _depth=0):
206 """Load a fulfillment for a signing spec dictionary
207
208 Args:
209 data: tx.output[].condition.details dictionary
210 """
211 if _depth == 100:
212 raise ThresholdTooDeep()
213
214 if data['type'] == 'ed25519-sha-256':
215 public_key = base58.b58decode(data['public_key'])
216 return Ed25519Sha256(public_key=public_key)
217
218 if data['type'] == 'threshold-sha-256':
219 threshold = ThresholdSha256(data['threshold'])
220 for cond in data['subconditions']:
221 cond = _fulfillment_from_details(cond, _depth+1)
222 threshold.add_subfulfillment(cond)
223 return threshold
224
225 raise UnsupportedTypeError(data.get('type'))

228class TransactionLink(object):
229 """An object for unidirectional linking to a Transaction's Output.
230
231 Attributes:
232 txid (str, optional): A Transaction to link to.
233 output (int, optional): An output's index in a Transaction with id
234 `txid`.
235 """
236
237 def __init__(self, txid=None, output=None):
238 """Create an instance of a :class:`~.TransactionLink`.
239
240 Note:
241 In an IPLD implementation, this class is not necessary anymore,
242 as an IPLD link can simply point to an object, as well as an
243 objects properties. So instead of having a (de)serializable
244 class, we can have a simple IPLD link of the form:
245 `/<tx_id>/transaction/outputs/<output>/`.
246
247 Args:
248 txid (str, optional): A Transaction to link to.
249 output (int, optional): An Outputs's index in a Transaction with
250 id `txid`.
251 """
252 self.txid = txid
253 self.output = output
254
255 def __bool__(self):
256 return self.txid is not None and self.output is not None
257
258 def __eq__(self, other):
259 # TODO: If `other !== TransactionLink` return `False`
260 return self.to_dict() == other.to_dict()
261
262 def __hash__(self):
263 return hash((self.txid, self.output))
264
265 @classmethod
266 def from_dict(cls, link):
267 """Transforms a Python dictionary to a TransactionLink object.
268
269 Args:
270 link (dict): The link to be transformed.
271
272 Returns:
273 :class:`~resdb_validator.transaction.TransactionLink`
274 """
275 try:
276 return cls(link['transaction_id'], link['output_index'])
277 except TypeError:
278 return cls()
279
280 def to_dict(self):
281 """Transforms the object to a Python dictionary.
282
283 Returns:
284 (dict|None): The link as an alternative serialization format.
285 """
286 if self.txid is None and self.output is None:
287 return None
288 else:
289 return {
290 'transaction_id': self.txid,
291 'output_index': self.output,
292 }

296 return None

298 self.output)

301class Output(object):
302 """An Output is used to lock an asset.
303
304 Wraps around a Crypto-condition Condition.
305
306 Attributes:
307 fulfillment (:class:`cryptoconditions.Fulfillment`): A Fulfillment
308 to extract a Condition from.
309 public_keys (:obj:`list` of :obj:`str`, optional): A list of
310 owners before a Transaction was confirmed.
311 """
312
313 MAX_AMOUNT = 9 * 10 ** 18
314
315 def __init__(self, fulfillment, public_keys=None, amount=1):
316 """Create an instance of a :class:`~.Output`.
317
318 Args:
319 fulfillment (:class:`cryptoconditions.Fulfillment`): A
320 Fulfillment to extract a Condition from.
321 public_keys (:obj:`list` of :obj:`str`, optional): A list of
322 owners before a Transaction was confirmed.
323 amount (int): The amount of Assets to be locked with this
324 Output.
325
326 Raises:
327 TypeError: if `public_keys` is not instance of `list`.
328 """
329 if not isinstance(public_keys, list) and public_keys is not None:
330 raise TypeError('`public_keys` must be a list instance or None')
331 if not isinstance(amount, int):
332 raise TypeError('`amount` must be an int')
333 if amount < 1:
334 raise AmountError('`amount` must be greater than 0')
335 if amount > self.MAX_AMOUNT:
336 raise AmountError('`amount` must be <= %s' % self.MAX_AMOUNT)
337
338 self.fulfillment = fulfillment
339 self.amount = amount
340 self.public_keys = public_keys
341
342 def __eq__(self, other):
343 # TODO: If `other !== Condition` return `False`
344 return self.to_dict() == other.to_dict()
345
346 def to_dict(self):
347 """Transforms the object to a Python dictionary.
348
349 Note:
350 A dictionary serialization of the Input the Output was
351 derived from is always provided.
352
353 Returns:
354 dict: The Output as an alternative serialization format.
355 """
356 # TODO FOR CC: It must be able to recognize a hashlock condition
357 # and fulfillment!
358 condition = {}
359 try:
360 condition['details'] = _fulfillment_to_details(self.fulfillment)
361 except AttributeError:
362 pass
363
364 try:
365 condition['uri'] = self.fulfillment.condition_uri
366 except AttributeError:
367 condition['uri'] = self.fulfillment
368
369 output = {
370 'public_keys': self.public_keys,
371 'condition': condition,
372 'amount': str(self.amount),
373 }
374 return output

376 @classmethod
377 def generate(cls, public_keys, amount):
378 """Generates a Output from a specifically formed tuple or list.
379
380 Note:
381 If a ThresholdCondition has to be generated where the threshold
382 is always the number of subconditions it is split between, a
383 list of the following structure is sufficient:
384
385 [(address|condition)*, [(address|condition)*, ...], ...]
386
387 Args:
388 public_keys (:obj:`list` of :obj:`str`): The public key of
389 the users that should be able to fulfill the Condition
390 that is being created.
391 amount (:obj:`int`): The amount locked by the Output.
392
393 Returns:
394 An Output that can be used in a Transaction.
395
396 Raises:
397 TypeError: If `public_keys` is not an instance of `list`.
398 ValueError: If `public_keys` is an empty list.
399 """
400 threshold = len(public_keys)
401 if not isinstance(amount, int):
402 raise TypeError('`amount` must be a int')
403 if amount < 1:
404 raise AmountError('`amount` needs to be greater than zero')
405 if not isinstance(public_keys, list):
406 raise TypeError('`public_keys` must be an instance of list')
407 if len(public_keys) == 0:
408 raise ValueError('`public_keys` needs to contain at least one'
409 'owner')
410 elif len(public_keys) == 1 and not isinstance(public_keys[0], list):
411 if isinstance(public_keys[0], Fulfillment):
412 ffill = public_keys[0]
413 else:
414 ffill = Ed25519Sha256(
415 public_key=base58.b58decode(public_keys[0]))
416 return cls(ffill, public_keys, amount=amount)
417 else:
418 initial_cond = ThresholdSha256(threshold=threshold)
419 threshold_cond = reduce(cls._gen_condition, public_keys,
420 initial_cond)
421 return cls(threshold_cond, public_keys, amount=amount)
422
423 @classmethod
424 def _gen_condition(cls, initial, new_public_keys):
425 """Generates ThresholdSha256 conditions from a list of new owners.
426
427 Note:
428 This method is intended only to be used with a reduce function.
429 For a description on how to use this method, see
430 :meth:`~.Output.generate`.
431
432 Args:
433 initial (:class:`cryptoconditions.ThresholdSha256`):
434 A Condition representing the overall root.
435 new_public_keys (:obj:`list` of :obj:`str`|str): A list of new
436 owners or a single new owner.
437
438 Returns:
439 :class:`cryptoconditions.ThresholdSha256`:
440 """
441 try:
442 threshold = len(new_public_keys)
443 except TypeError:
444 threshold = None
445
446 if isinstance(new_public_keys, list) and len(new_public_keys) > 1:
447 ffill = ThresholdSha256(threshold=threshold)
448 reduce(cls._gen_condition, new_public_keys, ffill)
449 elif isinstance(new_public_keys, list) and len(new_public_keys) <= 1:
450 raise ValueError('Sublist cannot contain single owner')
451 else:
452 try:
453 new_public_keys = new_public_keys.pop()
454 except AttributeError:
455 pass
456 # NOTE: Instead of submitting base58 encoded addresses, a user
457 # of this class can also submit fully instantiated
458 # Cryptoconditions. In the case of casting
459 # `new_public_keys` to a Ed25519Fulfillment with the
460 # result of a `TypeError`, we're assuming that
461 # `new_public_keys` is a Cryptocondition then.
462 if isinstance(new_public_keys, Fulfillment):
463 ffill = new_public_keys
464 else:
465 ffill = Ed25519Sha256(
466 public_key=base58.b58decode(new_public_keys))
467 initial.add_subfulfillment(ffill)
468 return initial

498class Transaction(object):
499 """A Transaction is used to create and transfer assets.
500
501 Note:
502 For adding Inputs and Outputs, this class provides methods
503 to do so.
504
505 Attributes:
506 operation (str): Defines the operation of the Transaction.
507 inputs (:obj:`list` of :class:`~resdb_validator.
508 transaction.Input`, optional): Define the assets to
509 spend.
510 outputs (:obj:`list` of :class:`~resdb_validator.
511 transaction.Output`, optional): Define the assets to lock.
512 asset (dict): Asset payload for this Transaction. ``CREATE``
513 Transactions require a dict with a ``data``
514 property while ``TRANSFER`` Transactions require a dict with a
515 ``id`` property.
516 metadata (dict):
517 Metadata to be stored along with the Transaction.
518 version (string): Defines the version number of a Transaction.
519 """
520
521 CREATE = 'CREATE'
522 TRANSFER = 'TRANSFER'
523 ALLOWED_OPERATIONS = (CREATE, TRANSFER)
524 VERSION = '2.0'
525
526 def __init__(self, operation, asset, inputs=None, outputs=None,
527 metadata=None, version=None, hash_id=None, tx_dict=None):
528 """The constructor allows to create a customizable Transaction.
529
530 Note:
531 When no `version` is provided, one is being
532 generated by this method.
533
534 Args:
535 operation (str): Defines the operation of the Transaction.
536 asset (dict): Asset payload for this Transaction.
537 inputs (:obj:`list` of :class:`~resdb_validator.
538 transaction.Input`, optional): Define the assets to
539 outputs (:obj:`list` of :class:`~resdb_validator.
540 transaction.Output`, optional): Define the assets to
541 lock.
542 metadata (dict): Metadata to be stored along with the
543 Transaction.
544 version (string): Defines the version number of a Transaction.
545 hash_id (string): Hash id of the transaction.
546 """
547 if operation not in self.ALLOWED_OPERATIONS:
548 allowed_ops = ', '.join(self.__class__.ALLOWED_OPERATIONS)
549 raise ValueError('`operation` must be one of {}'
550 .format(allowed_ops))
551
552 # Asset payloads for 'CREATE' operations must be None or
553 # dicts holding a `data` property. Asset payloads for 'TRANSFER'
554 # operations must be dicts holding an `id` property.
555 if (operation == self.CREATE and
556 asset is not None and not (isinstance(asset, dict) and 'data' in asset)):
557 raise TypeError(('`asset` must be None or a dict holding a `data` '
558 " property instance for '{}' Transactions".format(operation)))
559 elif (operation == self.TRANSFER and
560 not (isinstance(asset, dict) and 'id' in asset)):
561 raise TypeError(('`asset` must be a dict holding an `id` property '
562 'for \'TRANSFER\' Transactions'))
563
564 if outputs and not isinstance(outputs, list):
565 raise TypeError('`outputs` must be a list instance or None')
566
567 if inputs and not isinstance(inputs, list):
568 raise TypeError('`inputs` must be a list instance or None')
569
570 if metadata is not None and not isinstance(metadata, dict):
571 raise TypeError('`metadata` must be a dict or None')
572
573 self.version = version if version is not None else self.VERSION
574 self.operation = operation
575 self.asset = asset
576 self.inputs = inputs or []
577 self.outputs = outputs or []
578 self.metadata = metadata
579 self._id = hash_id
580 self.tx_dict = tx_dict

707 @classmethod
708 def transfer(cls, inputs, recipients, asset_id, metadata=None):
709 """A simple way to generate a `TRANSFER` transaction.
710
711 Note:
712 Different cases for threshold conditions:
713
714 Combining multiple `inputs` with an arbitrary number of
715 `recipients` can yield interesting cases for the creation of
716 threshold conditions we'd like to support. The following
717 notation is proposed:
718
719 1. The index of a `recipient` corresponds to the index of
720 an input:
721 e.g. `transfer([input1], [a])`, means `input1` would now be
722 owned by user `a`.
723
724 2. `recipients` can (almost) get arbitrary deeply nested,
725 creating various complex threshold conditions:
726 e.g. `transfer([inp1, inp2], [[a, [b, c]], d])`, means
727 `a`'s signature would have a 50% weight on `inp1`
728 compared to `b` and `c` that share 25% of the leftover
