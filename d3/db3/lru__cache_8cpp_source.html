40template <typename KeyType, typename ValueType>
41ValueType LRUCache<KeyType, ValueType>::Get(KeyType key) {
42 if (lookup_.find(key) == lookup_.end()) {
43 cache_misses_++;
44 return ValueType();
45 }
46
47 // Move accessed key to front of key list. This marks the key as used
48 key_list_.splice(key_list_.begin(), key_list_, rlookup_[key]);
49
50 cache_hits_++;
51 return lookup_[key];
52}
53
54template <typename KeyType, typename ValueType>
55void LRUCache<KeyType, ValueType>::Put(KeyType key, ValueType value) {
56 if (lookup_.find(key) == lookup_.end()) {
57 if (key_list_.size() == capacity_) {
58 KeyType lru_key = key_list_.back();
59 key_list_.pop_back();
60 lookup_.erase(lru_key);
61 rlookup_.erase(lru_key);
62 }
63 key_list_.push_front(key);
64 rlookup_[key] = key_list_.begin();
65 } else {
66 key_list_.splice(key_list_.begin(), key_list_, rlookup_[key]);
67 }
68 lookup_[key] = value; // Set the lookup_ here
69}

71template <typename KeyType, typename ValueType>
72void LRUCache<KeyType, ValueType>::SetCapacity(int new_capacity) {
73 if (new_capacity < capacity_) {
74 while (key_list_.size() > new_capacity) {
75 KeyType lru_key = key_list_.back();
76 key_list_.pop_back();
77 lookup_.erase(lru_key);
78 rlookup_.erase(lru_key);
79 }
80 }
81 capacity_ = new_capacity;
82}

108template <typename KeyType, typename ValueType>
109double LRUCache<KeyType, ValueType>::GetCacheHitRatio() const {
110 int total_accesses = cache_hits_ + cache_misses_;
111 if (total_accesses == 0) {
112 return 0.0;
113 }
114 return static_cast<double>(cache_hits_) / total_accesses;
115}
