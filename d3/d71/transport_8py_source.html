51 def forward_request(
52 self,
53 method: str,
54 path: str = None,
55 json: dict = None,
56 params: dict = None,
57 headers: dict = None,
58 ) -> Response.json:
59 """! Makes HTTP requests to the configured nodes.
60 Retries connection errors
61 (e.g. DNS failures, refused connection, etc).
62 A user may choose to retry other errors
63 by catching the corresponding
64 exceptions and retrying `forward_request`.
65 Exponential backoff is implemented individually for each node.
66 Backoff delays are expressed as timestamps stored on the object and
67 they are not reset in between multiple function calls.
68 Times out when `self.timeout` is expired, if not `None`.
69
70 @param method (str): HTTP method name (e.g.: ``'GET'``).
71 @param path (str): Path to be appended to the base url of a node. E.g.:
72 ``'/transactions'``).
73 @param json (dict): Payload to be sent with the HTTP request.
74 @param params (dict): Dictionary of URL (query) parameters.
75 @param headers (dict): Optional headers to pass to the request.
76
77 @return Result of :meth:`requests.models.Response.json`
78
79 """
80 error_trace = []
81 timeout = self.timeout
82 backoff_cap = NO_TIMEOUT_BACKOFF_CAP if timeout is None else timeout / 2
83 while timeout is None or timeout > 0:
84 connection: Connection = self.connection_pool.get_connection()
85
86 start = time()
87 try:
88 response = connection.request(
89 method=method,
90 path=path,
91 params=params,
92 json=json,
93 headers=headers,
94 timeout=timeout,
95 backoff_cap=backoff_cap,
96 )
97 except ConnectionError as err:
98 error_trace.append(err)
99 continue
100 else:
101 return response.data
102 finally:
103 elapsed = time() - start
104 if timeout is not None:
105 timeout -= elapsed
106
107 raise TimeoutError(error_trace)
