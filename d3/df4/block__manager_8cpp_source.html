40void BlockManager::SaveClientTransactions(
41 std::unique_ptr<BatchClientTransactions> client_request) {
42 for (const ClientTransactions& client_tx : client_request->transactions()) {
43 *request_candidate_.add_transactions() = client_tx;
44 }
45 if (request_candidate_.min_seq() == 0) {
46 request_candidate_.set_min_seq(client_request->min_seq());
47 } else {
48 request_candidate_.set_min_seq(
49 std::min(request_candidate_.min_seq(), client_request->min_seq()));
50 }
51 request_candidate_.set_max_seq(
52 std::max(request_candidate_.max_seq(), client_request->max_seq()));
53}

55int BlockManager::SetNewMiningBlock(
56 std::unique_ptr<BatchClientTransactions> client_request) {
57 SaveClientTransactions(std::move(client_request));
58
59 std::unique_ptr<Block> new_block = std::make_unique<Block>();
60 if (request_candidate_.min_seq() != GetLastSeq() + 1) {
61 LOG(ERROR) << "seq invalid:" << request_candidate_.min_seq()
62 << " last seq:" << GetLastSeq();
63 return -2;
64 }
65
66 int64_t new_time = 0;
67 for (ClientTransactions& client_tx :
68 *request_candidate_.mutable_transactions()) {
69 new_time = client_tx.create_time();
70 if (new_time > 0) {
71 create_time_[client_tx.seq()] = new_time;
72 client_tx.clear_create_time();
73 }
74 }
75
76 new_block->mutable_header()->set_height(GetCurrentHeight() + 1);
77 *new_block->mutable_header()->mutable_pre_hash() =
78 GetPreviousBlcokHash(); // set the hash value of the parent block.
79 request_candidate_.SerializeToString(new_block->mutable_transaction_data());
80 new_block->set_min_seq(request_candidate_.min_seq());
81 new_block->set_max_seq(request_candidate_.max_seq());
82 *new_block->mutable_header()->mutable_merkle_hash() =
83 Merkle::MakeHash(request_candidate_);
84 new_block->set_miner(config_.GetSelfInfo().id());
85 new_block->set_block_time(GetCurrentTime());
86
87 LOG(ERROR) << "create new block:"
88 << request_candidate_.transactions(0).create_time() << "["
89 << new_block->min_seq() << "," << new_block->max_seq() << "]"
90 << " miner:" << config_.GetSelfInfo().id()
91 << " time:" << new_block->block_time()
92 << " delay:" << (new_block->block_time() - new_time) / 1000000.0
93 << " current:" << GetCurrentTime();
94 new_mining_block_ = std::move(new_block);
95 miner_->SetSliceIdx(0);
96 return 0;
97}

103// Mine the nonce.
104absl::Status BlockManager::Mine() {
105 if (new_mining_block_ == nullptr) {
106 LOG(ERROR) << "don't contain mining block.";
107 return absl::InvalidArgumentError("height invalid");
108 }
109
110 if (new_mining_block_->header().height() != GetCurrentHeight() + 1) {
111 // a new block has been committed.
112 LOG(ERROR) << "new block height:" << new_mining_block_->header().height()
113 << " current height:" << GetCurrentHeight() << " not equal";
114 return absl::InvalidArgumentError("height invalid");
115 }
116
117 return miner_->Mine(new_mining_block_.get());
118}

122int BlockManager::SetSliceIdx(const SliceInfo& slice_info) {
123 if (new_mining_block_ == nullptr) {
124 return 0;
125 }
126 if (slice_info.height() != new_mining_block_->header().height()) {
127 return -2;
128 }
129 size_t f = config_.GetMaxMaliciousReplicaNum();
130 if (slice_info.shift_idx() > f) {
131 // reset
132 LOG(ERROR) << "reset slice";
133 miner_->SetSliceIdx(0);
134 return 1;
135 }
136 miner_->SetSliceIdx(slice_info.shift_idx());
137 return 0;
138}

140int BlockManager::Commit() {
141 request_candidate_.Clear();
142 uint64_t mining_time = GetCurrentTime() - new_mining_block_->block_time();
143 new_mining_block_->set_mining_time(mining_time);
144 int ret = AddNewBlock(std::move(new_mining_block_));
145 new_mining_block_ = nullptr;
146 return ret;
147}

154int BlockManager::AddNewBlock(std::unique_ptr<Block> new_block) {
155 {
156 std::unique_lock<std::mutex> lck(mtx_);
157 if (new_block->header().height() != GetCurrentHeightNoLock() + 1) {
158 // a new block has been committed.
159 LOG(ERROR) << "new block height:" << new_block->header().height()
160 << " current height:" << GetCurrentHeightNoLock()
161 << " not equal";
162 return -2;
163 }
164 LOG(INFO) << "============= commit new block:"
165 << new_block->header().height()
166 << " current height:" << GetCurrentHeightNoLock();
167 LOG(ERROR) << "commit:" << new_block->header().height()
168 << " from:" << new_block->miner()
169 << " mining time:" << new_block->mining_time() / 1000000.0;
170 // prometheus_handler_->SetValue(MINING_TIME,
171 // new_block->mining_time()/1000000.0);
172 // prometheus_handler_->Inc(MINING_LATENCY,
173 // new_block->mining_time()/1000000000.0);
174 miner_->Terminate();
175 request_candidate_.Clear();
176 block_list_.push_back(std::move(new_block));
177 Execute(*block_list_.back());
178 }
179 return 0;
180}

255bool BlockManager::VerifyBlock(const Block* block) {
256 if (!miner_->IsValidHash(block)) {
257 LOG(ERROR) << "hash not valid:" << block->hash().DebugString();
258 return false;
259 }
260
261 BatchClientTransactions client_request;
262 if (!client_request.ParseFromString(block->transaction_data())) {
263 LOG(ERROR) << "parse transaction fail";
264 return false;
265 }
266 return Merkle::MakeHash(client_request) == block->header().merkle_hash();
267}
