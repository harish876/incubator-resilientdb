44std::unique_ptr<std::string>
45KVServiceTransactionManager::ExecuteData(const std::string &request) {
46 KVRequest kv_request;
47 KVResponse kv_response;
48
49 if (!kv_request.ParseFromString(request)) {
50 LOG(ERROR) << "parse data fail";
51 return nullptr;
52 }
53
54 if (kv_request.cmd() == KVRequest::SET) {
55 Set(kv_request.key(), kv_request.value());
56 } else if (kv_request.cmd() == KVRequest::GET) {
57 kv_response.set_value(Get(kv_request.key()));
58 } else if (kv_request.cmd() == KVRequest::GETALLVALUES) {
59 kv_response.set_value(GetAllValues());
60 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
61 kv_response.set_value(GetRange(kv_request.key(), kv_request.value()));
62 }
63
64 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
65 if (!kv_response.SerializeToString(resp_str.get())) {
66 return nullptr;
67 }
68
69 return resp_str;
70}
