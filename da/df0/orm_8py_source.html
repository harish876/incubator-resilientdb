21class ResDBORM:
22 def __init__(self, config_path='config.yaml'):
23 with open(config_path, 'r') as config_file:

27 def generate_token(self, length=64):
28 """Generate a secure random hexadecimal token."""
29 return secrets.token_hex(length // 2)
30
31 def create(self, data):
32 """Create a new record in the DB."""
33 token = self.generate_token()
34 payload = {"id":token, "data":data}
35 headers = {'Content-Type': 'application/json'}
36 response = requests.post(f'{self.db_root_url}/v1/transactions/commit',
37 data=json.dumps(payload), headers=headers)
38
39
40 # Check if response is successful and handle empty response content
41 if response.status_code == 201:
42 if response.content:
43 decoded_content = response.content.decode('utf-8')
44 id_value = decoded_content.split(': ')[1].strip()
45 return id_value
46 else:
47 return {"status": "create unsuccessful, no content in response"}
48
49 def read_all(self):
50 """Read all records from the DB."""
51 response = requests.get(f'{self.db_root_url}/v1/transactions')
52 return response.json()

54 def read(self, key):

59 def delete(self, key):
60 """Delete a specific record by key in the DB."""
61 payload = {"id": key}
62 headers = {'Content-Type': 'application/json'}
63 response = requests.post(f'{self.db_root_url}/v1/transactions/commit',
64 data=json.dumps(payload), headers=headers)
65
66 # Check if response is successful and handle empty response content
67 if response.status_code == 201:

73 def update(self, key, new_data):
74 """Update a specific record by key in the DB."""
75 # Delete the existing record first

78 # Handle the response accordingly
