105 async def fetch_and_sync_initial_blocks(self):
106 try:
107 last_block = await self.collection.find_one({}, sort=[("id", -1)])
108 self.current_block_number = last_block['id'] if last_block and 'id' in last_block else 0
109
110 batch_size = 100
111 concurrency_limit = 5
112
113 batch_ranges = []
114 blocks_fetched = True
115
116 async with httpx.AsyncClient() as client:
117 while blocks_fetched:
118 min_seq = self.current_block_number + 1
119 max_seq = min_seq + batch_size - 1
120
121 url = f"{self.http_endpoint}/{min_seq}/{max_seq}"
122 logger.info(f"Checking for blocks from {min_seq} to {max_seq}")
123 response = await client.get(url)
124 blocks = response.json()
125
126 if isinstance(blocks, list) and blocks:
127 self.current_block_number = max(block.get('id', 0) for block in blocks)
128 batch_ranges.append({'min_seq': min_seq, 'max_seq': max_seq})
129 else:
130 blocks_fetched = False
131
132 if not batch_ranges:
133 logger.info("No new blocks to sync.")
134 return
135
136 logger.info(f"Total batches to fetch: {len(batch_ranges)}")
137
138 semaphore = asyncio.Semaphore(concurrency_limit)
139 tasks = [self.fetch_and_sync_batch(batch['min_seq'], batch['max_seq'], semaphore)
140 for batch in batch_ranges]
141 await asyncio.gather(*tasks)
142
143 except Exception as e:
144 logger.error("Error fetching initial blocks:")
145 logger.error(e)
146 raise ResilientPythonCacheError(str(e)) from e

148 async def fetch_and_sync_batch(self, min_seq: int, max_seq: int, semaphore: asyncio.Semaphore):
149 async with semaphore:
150 try:
151 async with httpx.AsyncClient() as client:
152 url = f"{self.http_endpoint}/{min_seq}/{max_seq}"
153 logger.info(f"Fetching blocks from {min_seq} to {max_seq}")
154 response = await client.get(url)
155 blocks = response.json()
156
157 if isinstance(blocks, list) and blocks:
158 processed_blocks = self.process_blocks(blocks)
159 bulk_ops = [
160 UpdateOne(
161 {'id': block['id']},
162 {'$set': block},
163 upsert=True
164 ) for block in processed_blocks
165 ]
166
167 if bulk_ops:
168 result = await self.collection.bulk_write(bulk_ops)
169 logger.info(f"Blocks {min_seq} to {max_seq} synced: "
170 f"Inserted {result.upserted_count}, Modified {result.modified_count}")
171 self.emit('data', processed_blocks) # Emit 'data' event with new blocks
172 else:
173 logger.info(f"No blocks fetched for range {min_seq} to {max_seq}")
174 except Exception as e:
175 logger.error(f"Error fetching blocks from {min_seq} to {max_seq}:")
176 logger.error(e)

178 def process_blocks(self, blocks: list) -> list:
179 for block in blocks:
180 transactions = block.get('transactions', [])
181 if isinstance(transactions, list):
182 for transaction in transactions:
183 if 'value' in transaction and isinstance(transaction['value'], str):
184 value = transaction['value'].strip()
185 if (value.startswith('{') and value.endswith('}')) or \
186 (value.startswith('[') and value.endswith(']')):
187 try:
188 transaction['value'] = json.loads(value)
189 except json.JSONDecodeError:
190 logger.warning(f"Could not parse transaction value for key {transaction.get('key')}. "
191 f"Keeping original value.")
192 return blocks

194 async def start_periodic_fetch(self):
195 logger.info(f"Started periodic fetch every {self.fetch_interval / 1000} seconds.")
196 try:
197 while not self.is_closing:
198 await self.fetch_and_sync_new_blocks()
199 await asyncio.sleep(self.fetch_interval / 1000)

201 pass # Allow task to be cancelled gracefully

242 async def connect_websocket(self):
243 try:
244 async with websockets.connect(self.ws_endpoint) as websocket:
245 logger.info(f"Connected to WebSocket: {self.ws_endpoint}")
246 self.reconnect_attempts = 0
247 self.emit('connected')
248 # Removed the duplicate log statement here
249 # logger.info("WebSocket connected.")
250
251 async for message in websocket:
252 logger.info(f"Received message: {message}")
253 try:
254 parsed_message = json.loads(message)
255 except json.JSONDecodeError:
256 parsed_message = {"type": message}
257
258 if parsed_message.get("type") == "Update blocks":
259 await self.fetch_and_sync_new_blocks()
260 else:
261 logger.warning(f"Received unrecognized message: {message}")
262 except (websockets.exceptions.ConnectionClosedError,
263 websockets.exceptions.InvalidURI,
264 websockets.exceptions.InvalidHandshake) as e:
265 logger.warning(f"WebSocket closed or failed to connect: {e}")
266 if not self.is_closing:
267 await self.attempt_reconnection()
268 except asyncio.CancelledError:
269 pass # Allow task to be cancelled gracefully
270 except Exception as e:
271 logger.error("WebSocket error:")
272 logger.error(e)
273 if not self.is_closing:
274 await self.attempt_reconnection()

276 async def attempt_reconnection(self):
277 delay = self.reconnect_interval * (2 ** self.reconnect_attempts) / 1000 # Convert ms to seconds
278 logger.info(f"Attempting to reconnect in {delay} seconds...")
279 await asyncio.sleep(delay)
280 self.reconnect_attempts += 1
281 logger.info(f"Reconnection attempt #{self.reconnect_attempts}")
282 self.ws_task = asyncio.create_task(self.connect_websocket())

284 async def close(self):
285 try:
286 self.is_closing = True
