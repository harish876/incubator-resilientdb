28def serialize(data: dict) -> str:
29 """! Serialize a dict into a JSON formatted string.
30
31 This function enforces rules like the separator and order of keys.
32 This ensures that all dicts are serialized in the same way.
33
34 This is specially important for hashing data. We need to make sure that
35 everyone serializes their data in the same way so that we do not have
36 hash mismatches for the same structure due to serialization
37 differences.
38
39 @param data (dict): Data to serialize
40
41 @return JSON formatted string
42
43 """
44 return rapidjson.dumps(data, skipkeys=False, ensure_ascii=False, sort_keys=True)

74def _normalize_operation(operation):
75 """! Normalizes the given operation string. For now, this simply means
76 converting the given string to uppercase, looking it up in
77 :attr:`~.ops_map`, and returning the corresponding class if
78 present.
79
80 @param operation (str): The operation string to convert.
81
82 @return The class corresponding to the given string,
83 :class:`~.CreateOperation` or :class:`~TransferOperation`.
84
85 .. important:: If the :meth:`str.upper` step, or the
86 :attr:`~.ops_map` lookup fails, the given ``operation``
87 argument is returned.
88
89 """
90 try:
91 operation = operation.upper()
92 except AttributeError:
93 pass
94
95 try:
96 operation = ops_map[operation]()
97 except KeyError:
98 pass
99
100 return operation

107def normalize_url(node):
108 """! Normalizes the given node url"""
109 if not node:
110 node = DEFAULT_NODE
111 elif "://" not in node:
112 node = "//{}".format(node)
113 parts = urlparse(node, scheme="http", allow_fragments=False)
114 port = parts.port if parts.port else _get_default_port(parts.scheme)
115 netloc = "{}:{}".format(parts.hostname, port)
116 return urlunparse((parts.scheme, netloc, parts.path, "", "", ""))

119def normalize_node(node, headers=None):
120 """! Normalizes given node as str or dict with headers"""
121 headers = {} if headers is None else headers
122 if isinstance(node, str):
123 url = normalize_url(node)
124 return {"endpoint": url, "headers": headers}
125
126 url = normalize_url(node["endpoint"])
127 node_headers = node.get("headers", {})
128 return {"endpoint": url, "headers": {**headers, **node_headers}}

131def normalize_nodes(*nodes, headers=None):
132 """! Normalizes given dict or array of driver nodes"""
133 if not nodes:
134 return (normalize_node(DEFAULT_NODE, headers),)
135
136 normalized_nodes = ()
137 for node in nodes:
138 normalized_nodes += (normalize_node(node, headers),)
139 return normalized_nodes
