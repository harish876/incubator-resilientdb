242 impl_ = std::make_unique<MockConsensusManager>(config_);
243 impl_->Start();
244
245 EXPECT_CALL(*impl_, GetReplicaClient)
246 .WillRepeatedly(

248 EXPECT_EQ(replicas.size(), replicas_.size());
249 auto client = std::make_unique<MockReplicaCommunicator>(replicas);
250 return client;
251 }));
252
253 HeartBeatInfo hb;
254 auto* public_key = hb.add_public_keys();
255
256 public_key->mutable_public_key_info()->set_ip("127.0.0.1");
257 public_key->mutable_public_key_info()->set_port(12345);
258 public_key->mutable_public_key_info()->set_node_id(5);
259 public_key->mutable_public_key_info()->set_type(CertificateKeyInfo::CLIENT);
260

262 request->set_type(Request::TYPE_HEART_BEAT);
263 hb.SerializeToString(request->mutable_data());
264
265 EXPECT_EQ(impl_->Dispatch(std::make_unique<Context>(), std::move(request)),
266 0);
267

269 EXPECT_NE(bc_client, nullptr);
