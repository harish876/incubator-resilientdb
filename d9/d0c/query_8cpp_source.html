40 if (context != nullptr && context->client != nullptr) {

43 LOG(ERROR) << "send resp" << replica_state.DebugString()

47 }

51int Query::ProcessQuery(std::unique_ptr<Context> context,
52 std::unique_ptr<Request> request) {
53 if (config_.GetPublicKeyCertificateInfo()
54 .public_key()
55 .public_key_info()
56 .type() == CertificateKeyInfo::CLIENT) {
57 auto find_primary = [&]() {
58 auto config_data = config_.GetConfigData();
59 for (const auto& r : config_data.region()) {
60 for (const auto& replica : r.replica_info()) {
61 if (replica.id() == 1) {
62 return replica;
63 }
64 }
65 }
66 };
67 ReplicaInfo primary = find_primary();
68 std::string ip = primary.ip();
69 int port = primary.port();
70
71 LOG(ERROR) << "redirect to primary:" << ip << " port:" << port;
72 auto client = std::make_unique<NetChannel>(ip, port);
73 if (client->SendRawMessage(*request) == 0) {
74 QueryResponse resp;
75 if (client->RecvRawMessage(&resp) == 0) {
76 if (context != nullptr && context->client != nullptr) {
77 LOG(ERROR) << "send response from primary:"
78 << resp.transactions_size();
79 int ret = context->client->SendRawMessage(resp);
80 if (ret) {
81 LOG(ERROR) << "send resp fail ret:" << ret;
82 }
83 }
84 }
85 }
86 return 0;
87 }
88
89 QueryRequest query;
90 if (!query.ParseFromString(request->data())) {
91 LOG(ERROR) << "parse data fail";
92 return -2;
93 }
94
95 QueryResponse response;
96 if (query.max_seq() == 0 && query.min_seq() == 0) {
97 uint64_t mseq = message_manager_->GetNextSeq();
98 response.set_max_seq(mseq - 1);
99 LOG(ERROR) << "get max seq:" << mseq;
100 } else {
101 for (uint64_t i = query.min_seq(); i <= query.max_seq(); ++i) {
102 Request* ret_request = message_manager_->GetRequest(i);
103 if (ret_request == nullptr) {
104 break;
105 }
106 Request* txn = response.add_transactions();
107 txn->set_data(ret_request->data());
108 txn->set_hash(ret_request->hash());
109 txn->set_seq(ret_request->seq());
110 txn->set_proxy_id(ret_request->proxy_id());
111 }
112 }
113
114 if (context != nullptr && context->client != nullptr) {
115 // LOG(ERROR) << "send response:" << response.DebugString();
116 int ret = context->client->SendRawMessage(response);
117 if (ret) {
118 LOG(ERROR) << "send resp fail ret:" << ret;
119 }
120 }
121 return 0;
122}

124int Query::ProcessCustomQuery(std::unique_ptr<Context> context,
125 std::unique_ptr<Request> request) {
126 if (custom_query_executor_ == nullptr) {
127 LOG(ERROR) << "no custom executor";
128 return -1;
129 }
130
131 std::unique_ptr<std::string> resp_str =
132 custom_query_executor_->Query(request->data());
133
134 CustomQueryResponse response;
135 if (resp_str != nullptr) {
136 response.set_resp_str(*resp_str);
137 }
138
139 if (context != nullptr && context->client != nullptr) {
140 int ret = context->client->SendRawMessage(response);
141 if (ret) {
142 LOG(ERROR) << "send resp fail ret:" << ret;
143 }
144 }
145 return 0;
146}
