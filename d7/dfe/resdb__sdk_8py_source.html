36class Transaction:
37 def __init__(self, id: str, value: str, metadata: Optional[Union[Dict, TransactionMetadata]] = None):
38 self.id = str(id)
39 self.value = str(value)
40 self.type = "kv"
41 self.metadata = metadata if isinstance(metadata, TransactionMetadata) else TransactionMetadata(**(metadata or {}))
42
43 def to_dict(self) -> Dict[str, Any]:
44 data = {
45 "id": self.id,
46 "value": self.value,
47 "type": self.type
48 }
49 if self.metadata and self.metadata.data:
50 data["metadata"] = self.metadata.to_dict()
51 return data

53class TransactionAPI:
54 def __init__(self, url: str):
55 self.url = url.rstrip("/")

57 async def create(self, transaction: Transaction) -> Dict[str, Any]:
58 """Create a new transaction"""
59 try:
60 headers = {
61 "Content-Type": "application/json",
62 "Accept": "*/*"
63 }
64
65 response = await pyfetch(
66 f"{self.url}/v1/transactions/commit",
67 method="POST",
68 headers=headers,
69 body=json.dumps(transaction.to_dict())
70 )
71
72 response_text = await response.text()
73
74 if response.status == 201 and response_text.startswith('id:'):
75 tx_id = response_text.split(':')[1].strip()
76 return {
77 "status": 201,
78 "success": True,
79 "id": tx_id,
80 "message": "Transaction created successfully"
81 }
82
83 if response.status == 200:
84 return {
85 "status": 200,
86 "success": True,
87 "id": transaction.id,
88 "message": "Transaction accepted"
89 }
90
91 return {
92 "status": response.status,
93 "success": False,
94 "error": response_text or "Transaction failed"
95 }
96
97 except Exception as e:
98 print(f"Error creating transaction: {str(e)}")
99 raise Exception(str(e))
100
101 async def retrieve(self, transaction_id: str) -> Dict[str, Any]:
102 """Retrieve a transaction by ID"""
103 try:
104 headers = {
105 "Accept": "application/json"
106 }
107
108 response = await pyfetch(
109 f"{self.url}/v1/transactions/{transaction_id}",
110 method="GET",
111 headers=headers
112 )
113
114 response_text = await response.text()
115
116 if response.status == 200:
117 if response_text:
118 try:
119 return json.loads(response_text)
120 except:
121 return {
122 "status": 200,
123 "success": True,
124 "id": transaction_id,
125 "value": response_text.strip(),
126 "type": "kv"
127 }
128 return {
129 "status": 404,
130 "success": False,
131 "message": "Transaction not found (empty response)"
132 }
133
134 return {
135 "status": response.status,
136 "success": False,
137 "message": "Transaction not found"
138 }
139
140 except Exception as e:
141 print(f"Error retrieving transaction: {str(e)}")
142 raise Exception(str(e))

144class ResilientDB:
145 def __init__(self, url: str = RESDB_REST_URL):
146 """Initialize ResilientDB client"""
147 self.url = url.rstrip("/")
148 self.transactions = TransactionAPI(self.url)
149
150 async def get_info(self) -> Dict[str, Any]:
151 """Get ResilientDB node information"""
152 try:
153 response = await pyfetch(
154 f"{self.url}/v1/info",
155 method="GET",
156 headers={"Accept": "application/json"}
157 )
158
159 if response.status == 200:
160 return await response.json()
161 return {"error": "Failed to get node info"}
162
163 except Exception as e:
164 return {"error": f"Failed to get node info: {str(e)}"}
