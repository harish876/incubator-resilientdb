38class RoundRobinPicker(AbstractPicker):
39 """! Picks a :class:`~resdb_driver.connection.Connection`
40 instance from a list of connections.
41 """
42
43 def pick(self, connections: list[Connection]) -> Connection:
44 """! Picks a connection with the earliest backoff time.
45 As a result, the first connection is picked
46 for as long as it has no backoff time.
47 Otherwise, the connections are tried in a round robin fashion.
48
49 @param connections (:obj:list): List of :class:`~resdb_driver.connection.Connection` instances.
50 """
51 if len(connections) == 1:
52 return connections[0]
53
54 return min(
55 *connections,
56 key=lambda conn: datetime.min
57 if conn.backoff_time is None
58 else conn.backoff_time
59 )
