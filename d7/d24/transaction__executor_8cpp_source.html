27TransactionExecutor::TransactionExecutor(
28 const ResDBConfig& config, PostExecuteFunc post_exec_func,
29 SystemInfo* system_info,
30 std::unique_ptr<TransactionManager> transaction_manager)
31 : config_(config),
32 post_exec_func_(post_exec_func),
33 system_info_(system_info),
34 transaction_manager_(std::move(transaction_manager)),
35 commit_queue_("order"),
36 execute_queue_("execute"),
37 stop_(false),
38 duplicate_manager_(nullptr) {
39
40 memset(blucket_, 0, sizeof(blucket_));
41 global_stats_ = Stats::GetGlobalStats();
42 ordering_thread_ = std::thread(&TransactionExecutor::OrderMessage, this);
43 for (int i = 0; i < execute_thread_num_; ++i) {
44 execute_thread_.push_back(

54 execute_OOO_thread_ =
55 std::thread(&TransactionExecutor::ExecuteMessageOutOfOrder, this);
56 LOG(ERROR) << " is out of order:" << transaction_manager_->IsOutOfOrder();
57 }

62void TransactionExecutor::RegisterExecute(int64_t seq) {
63 if (execute_thread_num_ == 1) return;
64 int idx = seq % blucket_num_;
65 std::unique_lock<std::mutex> lk(mutex_);
66 // LOG(ERROR)<<"register seq:"<<seq<<" bluck:"<<blucket_[idx];
67 assert(!blucket_[idx] || !(blucket_[idx] ^ 3));
68 blucket_[idx] = 1;
69 // LOG(ERROR)<<"register seq:"<<seq;
70}

72void TransactionExecutor::WaitForExecute(int64_t seq) {
73 if (execute_thread_num_ == 1) return;
74 int pre_idx = (seq - 1 + blucket_num_) % blucket_num_;
75
76 while (!IsStop()) {
77 std::unique_lock<std::mutex> lk(mutex_);
78 cv_.wait_for(lk, std::chrono::milliseconds(10000), [&] {
79 return ((blucket_[pre_idx] & 2) || !blucket_[pre_idx]);
80 });
81 if ((blucket_[pre_idx] & 2) || !blucket_[pre_idx]) {
82 break;
83 }
84 }
85 // LOG(ERROR)<<"wait for :"<<seq<<" done";
86}

88void TransactionExecutor::FinishExecute(int64_t seq) {

170void TransactionExecutor::OrderMessage() {
171 while (!IsStop()) {
172 auto message = commit_queue_.Pop();
173 if (message != nullptr) {
174 global_stats_->IncExecute();
175 uint64_t seq = message->seq();
176 if (next_execute_seq_ > seq) {
177 // LOG(INFO) << "request seq:" << seq << " has been executed"
178 // << " next seq:" << next_execute_seq_;
179 continue;
180 }
181
182 AddNewData(std::move(message));
183 }
184
185 while (!IsStop()) {
186 std::unique_ptr<Request> message = GetNextData();
187 if (message == nullptr) {
188 break;
189 }
190 execute_queue_.Push(std::move(message));
191 next_execute_seq_++;
192 if (seq_update_notify_func_) {
193 seq_update_notify_func_(next_execute_seq_);
194 }
195 }
196 }
197 return;
198}

256void TransactionExecutor::Execute(std::unique_ptr<Request> request,
257 bool need_execute) {
258 RegisterExecute(request->seq());
259 std::unique_ptr<BatchUserRequest> batch_request = nullptr;
260 std::unique_ptr<std::vector<std::unique_ptr<google::protobuf::Message>>> data;
261 std::vector<std::unique_ptr<google::protobuf::Message>> * data_p = nullptr;
262 BatchUserRequest* batch_request_p = nullptr;
263
264 // Execute the request, then send the response back to the user.
265 if (batch_request_p == nullptr) {
266 batch_request = std::make_unique<BatchUserRequest>();
267 if (!batch_request->ParseFromString(request->data())) {
268 LOG(ERROR) << "parse data fail";
269 }
270 batch_request->set_hash(request->hash());
271 if (request->has_committed_certs()) {
272 *batch_request->mutable_committed_certs() = request->committed_certs();
273 }
274 batch_request->set_seq(request->seq());
275 batch_request->set_proxy_id(request->proxy_id());
276 batch_request_p = batch_request.get();
277 // LOG(ERROR)<<"get data from req:";
278 } else {
279 assert(batch_request_p);
280 batch_request_p->set_seq(request->seq());
281 batch_request_p->set_proxy_id(request->proxy_id());
282 // LOG(ERROR)<<" get from cache:"<<uid;
283 }
284 assert(batch_request_p);
285
286 // LOG(INFO) << " get request batch size:"
287 // << batch_request.user_requests_size()<<" proxy id:"
288 // <<request->proxy_id()<<" need execute:"<<need_execute;
289
290 std::unique_ptr<BatchUserResponse> response;
291 global_stats_->GetTransactionDetails(*batch_request_p);
292 if (transaction_manager_ && need_execute) {
293 if (execute_thread_num_ == 1) {
294 response = transaction_manager_->ExecuteBatch(*batch_request_p);
295 } else {
296 std::vector<std::unique_ptr<std::string>> response_v;
297
298 if(data_p == nullptr) {
299 int64_t start_time = GetCurrentTime();
300 data = std::move(transaction_manager_->Prepare(*batch_request_p));
301 int64_t end_time = GetCurrentTime();
302 if (end_time - start_time > 10) {
303 // LOG(ERROR)<<"exec data done:"<<uid<<" wait
304 // time:"<<(end_time-start_time);
305 }
306 data_p = data.get();
307 }
308
309 WaitForExecute(request->seq());
310 if(data_p->empty() || (*data_p)[0] == nullptr){
311 response = transaction_manager_->ExecuteBatch(*batch_request_p);
312 }
313 else {
314 response_v = transaction_manager_->ExecuteBatchData(*data_p);
315 }
316 FinishExecute(request->seq());
317
318 if(response == nullptr){
319 response = std::make_unique<BatchUserResponse>();
320 for (auto& s : response_v) {
321 response->add_response()->swap(*s);
322 }
323 }
324 }
325 }
326 // LOG(ERROR)<<" CF = :"<<(cf==1)<<" uid:"<<uid;
327
328 if (duplicate_manager_ && batch_request_p) {
329 duplicate_manager_->AddExecuted(batch_request_p->hash(), batch_request_p->seq());
330 }
331
332 if (response == nullptr) {
333 response = std::make_unique<BatchUserResponse>();
334 }
335 global_stats_->IncTotalRequest(batch_request_p->user_requests_size());
336 response->set_proxy_id(batch_request_p->proxy_id());
337 response->set_createtime(batch_request_p->createtime() + request->queuing_time());
338 response->set_local_id(batch_request_p->local_id());
339
340 response->set_seq(request->seq());
341
342 if (post_exec_func_) {
343 post_exec_func_(std::move(request), std::move(response));
344 }
345
346 global_stats_->IncExecuteDone();
347}
