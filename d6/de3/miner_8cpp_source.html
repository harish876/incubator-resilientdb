53void Miner::SetSliceIdx(int slice_idx) {
54 shift_idx_ = slice_idx;
55
56 // the maximun bits used for mining.
57 uint64_t total_slice = config_.GetMaxNonceBit();
58 assert(total_slice <= 64);
59 total_slice = 1ll << total_slice;
60
61 size_t replica_size = config_.GetReplicaNum();
62
63 // id starts from 1.
64 uint32_t replica_id = config_.GetSelfInfo().id();
65 assert(total_slice >= replica_size);
66
67 int idx = (replica_id - 1 + shift_idx_) % replica_size + 1;
68
69 // the default slice is [min_slice, max_slice].
70 uint64_t min_slice = 0;
71 uint64_t max_slice = total_slice - 1;
72 LOG(ERROR) << "slice idx ??:" << slice_idx
73 << " total slice size:" << total_slice
74 << " replica id:" << replica_id << " replica size:" << replica_size
75 << " idx:" << idx << " min slice:" << min_slice
76 << " max slice:" << max_slice;
77 mining_slices_.clear();
78 mining_slices_.push_back(std::make_pair(min_slice, max_slice));
79}

81absl::Status Miner::Mine(Block* new_block) {
82 if (new_block->header().height() == 10 &&
83 new_block->max_seq() - new_block->min_seq() + 1 <=
84 config_.BatchTransactionNum() &&
85 shift_idx_ == 0) {
86 LOG(ERROR) << "skip fake fail";
87 return absl::NotFoundError("solution not found");
88 }
89 if (new_block->header().height() == 30 &&
90 new_block->max_seq() - new_block->min_seq() + 1 <=
91 config_.BatchTransactionNum() &&
92 shift_idx_ == 0) {
93 LOG(ERROR) << "skip fake fail";
94 return absl::NotFoundError("solution not found");
95 }
96 LOG(ERROR) << " start mine block slice:" << shift_idx_;
97 stop_ = false;
98 std::vector<std::pair<uint64_t, uint64_t>> slices = GetMiningSlices();
99
100 Block::Header header(new_block->header());
101
102 for (const auto& slice : slices) {
103 uint64_t max_slice = slice.second;
104 uint64_t min_slice = slice.first;
105
106 std::vector<std::thread> ths;
107 std::atomic<bool> solution_found = false;
108
109 uint64_t step = static_cast<uint64_t>(worker_num_);
110 for (uint32_t i = 0; i < worker_num_; ++i) {
111 uint64_t current_slice_start = i + min_slice;
112
113 ths.push_back(std::thread(
114 [&](Block::Header header, std::pair<uint64_t, uint64_t> slice) {
115 std::string header_hash;
116 header_hash += GetHashDigest(header.pre_hash());
117 header_hash += GetHashDigest(header.merkle_hash());
118
119 for (uint64_t nonce = slice.first;
120 nonce <= max_slice && !stop_ && !solution_found;
121 nonce += step) {
122 // LOG(ERROR)<<" miner min:"<<slice.first<<" max:"<<max_slice<<"
123 // nonce"<<nonce;
124 header.set_nonce(nonce);
125 std::string str_value =
126 header_hash + std::to_string(header.nonce());
127 std::string hash_digest = GetHashValue(str_value);
128
129 if (IsValidDigest(hash_digest, difficulty_)) {
130 solution_found = true;
131 *new_block->mutable_hash() = DigestToHash(hash_digest);
132 *new_block->mutable_header() = header;
133 LOG(ERROR) << "nonce:" << nonce
134 << " hex string:" << GetDigestHexString(hash_digest)
135 << " target:" << difficulty_;
136 return;
137 }
138 }
139 LOG(ERROR) << "mine done slice:" << slice.first << " len:" << step;
140 },
141 header, std::make_pair(current_slice_start, 0)));
142 }
143
144 for (auto& th : ths) {
145 th.join();
146 }
147 if (stop_) {
148 LOG(ERROR) << "minning has been terminated.";
149 return absl::CancelledError("terminated");
150 }
151
152 if (solution_found) {
153 LOG(ERROR) << "find solution:" << new_block->header().DebugString()
154 << " hashvalue:" << new_block->hash().DebugString();
155 return absl::OkStatus();
156 }
157 }
158 LOG(ERROR) << "solution not found";
159 return absl::NotFoundError("solution not found");
160}

167// Calculate the hash value: SHA256(SHA256(header))
168// The hash value will be a 32bit integer.
169std::string Miner::CalculatePoWHashDigest(const Block::Header& header) {
170 std::string str_value;
171 str_value += GetHashDigest(header.pre_hash());
172 str_value += GetHashDigest(header.merkle_hash());
173 str_value += std::to_string(header.nonce());
174 return SignatureVerifier::CalculateHash(
175 SignatureVerifier::CalculateHash(str_value));
176}

178HashValue Miner::CalculatePoWHash(const Block* new_block) {
179 std::string digest = CalculatePoWHashDigest(new_block->header());
180 return DigestToHash(digest);
