56def key_pair_from_ed25519_key(hex_private_key):
57 """Generate base58 encode public-private key pair from a hex encoded private key"""
58 priv_key = crypto.Ed25519SigningKey(bytes.fromhex(hex_private_key)[:32], encoding='bytes')
59 public_key = priv_key.get_verifying_key()
60 return CryptoKeypair(private_key=priv_key.encode(encoding='base58').decode('utf-8'),
61 public_key=public_key.encode(encoding='base58').decode('utf-8'))
