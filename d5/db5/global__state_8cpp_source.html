34uint256_t AccountToAddress(const eevm::Address& account) {
35 std::vector<uint8_t> code;
36 code.resize(64);
37 std::fill(code.begin(), code.end(), 0);
38 eevm::to_big_endian(account, code.data());
39
40 uint8_t h[32];
41 eevm::keccak_256(code.data(), static_cast<unsigned int>(64), h);
42 return eevm::from_big_endian(h, sizeof(h));
43}
