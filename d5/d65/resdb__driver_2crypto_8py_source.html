28def generate_keypair(seed=None) -> CryptoKeypair:
29 """! Generates a cryptographic key pair.
30
31 @param seed (bytes): 32-byte seed for deterministic generation. Defaults to `None`.
32
33 @return collections.namedtuple object containing the public and private keys.
34 """
35
36 return CryptoKeypair(*(k.decode() for k in crypto.ed25519_generate_key_pair(seed)))
