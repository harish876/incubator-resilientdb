19"""Utility script to generate Ed25519 keypairs for ResilientDB.
20
21This is a standalone utility script for manual key generation.
22For automated key generation, use the MCP server's generateKeys tool instead.
23"""
24import sys
25import os
26
27# Get the directory where this script is located
28script_dir = os.path.dirname(os.path.abspath(__file__))
29
30# Try to find the ResilientDB resdb_driver path
31# From ecosystem/mcp/ to ecosystem/graphql/resdb_driver
32possible_paths = [
33 # Relative path from mcp to graphql/resdb_driver (go up 1 level to ecosystem/)
34 os.path.join(script_dir, '../graphql/resdb_driver'),
35 # Absolute path (fallback)
36 '/Users/rahul/data/workspace/kanagrah/incubator-resilientdb/ecosystem/graphql/resdb_driver',
37]
38
39resilientdb_path = None
40for path in possible_paths:
41 abs_path = os.path.abspath(path)
42 if os.path.exists(abs_path):
43 sys.path.insert(0, abs_path)
44 resilientdb_path = abs_path
45 break
46
47if resilientdb_path is None:
48 print("Error: Could not find ResilientDB resdb_driver directory.")
49 print("Tried the following paths:")
50 for path in possible_paths:
51 print(f" - {os.path.abspath(path)}")
52 print(f"\nCurrent script location: {script_dir}")
53 sys.exit(1)
54
55try:
56 from crypto import generate_keypair
57except ImportError:
58 print("Error: Could not import generate_keypair from ResilientDB crypto module.")
59 print(f"Found path: {resilientdb_path}")
60 print(f"Please ensure the crypto.py file exists in: {resilientdb_path}")
61 sys.exit(1)
62
63# Generate keypairs
64signer = generate_keypair()
65recipient = generate_keypair()
66
67print("=" * 70)
68print("ResilientDB Key Generator")
69print("=" * 70)
70print()
71print("Signer Keypair:")
72print(f" Public Key: {signer.public_key}")
73print(f" Private Key: {signer.private_key}")
74print()
75print("Recipient Keypair:")
76print(f" Public Key: {recipient.public_key}")
77print(f" Private Key: {recipient.private_key}")
78print()
79print("=" * 70)
80print("Ready-to-use curl command:")
81print("=" * 70)
82print()
83print(f"""curl -X POST http://localhost:8000/graphql \\
84 -H "Content-Type: application/json" \\
85 -d '{{
86 "query": "mutation Test($data: PrepareAsset!) {{ postTransaction(data: $data) {{ id }} }}",
87 "variables": {{
88 "data": {{
89 "operation": "CREATE",
90 "amount": 100,
91 "signerPublicKey": "{signer.public_key}",
92 "signerPrivateKey": "{signer.private_key}",
93 "recipientPublicKey": "{recipient.public_key}",
94 "asset": {{
95 "data": {{
96 "name": "Test Asset",
97 "description": "My first test asset"
98 }}
99 }}
100 }}
101 }}
102 }}' | python3 -m json.tool""")
103print()
104print("=" * 70)
105print("Copy and paste the curl command above to create a transaction!")
106print("=" * 70)
