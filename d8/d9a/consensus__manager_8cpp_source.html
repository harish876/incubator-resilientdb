86// Keep Boardcast the public keys to others.
87void ConsensusManager::HeartBeat() {
88 LOG(INFO) << "heart beat start";
89 int sleep_time = 1;
90 std::mutex mutex;
91 std::condition_variable cv;
92 while (IsRunning()) {
93 {
94 std::unique_lock<std::mutex> lk(hb_mutex_);
95 SendHeartBeat();
96 }
97 std::unique_lock<std::mutex> lk(mutex);
98 cv.wait_for(lk, std::chrono::microseconds(sleep_time * 1000000),
99 [&] { return !IsRunning(); });
100 if (is_ready_) {
101 if (config_.IsTestMode()) {
102 sleep_time = 1;
103 } else {
104 sleep_time = 60;
105 }
106 }
107 }
108}

110void ConsensusManager::SendHeartBeat() {
111 auto keys = verifier_->GetAllPublicKeys();
112
113 std::vector<ReplicaInfo> replicas = GetAllReplicas();
114 LOG(ERROR) << "all replicas:" << replicas.size();
115 std::vector<ReplicaInfo> client_replicas = GetClientReplicas();
116 HeartBeatInfo hb_info;
117 hb_info.set_sender(config_.GetSelfInfo().id());
118 hb_info.set_ip(config_.GetSelfInfo().ip());
119 hb_info.set_port(config_.GetSelfInfo().port());
120 hb_info.set_hb_version(version_);
121 for (const auto& key : keys) {
122 *hb_info.add_public_keys() = key;
123 hb_info.add_node_version(hb_[key.public_key_info().node_id()]);
124 }
125 for (const auto& client : client_replicas) {
126 replicas.push_back(client);
127 }
128 auto client = GetReplicaClient(replicas, false);
129 if (client == nullptr) {
130 return;
131 }
132
133 // If it is not a client node, broadcost the current primary to the client.
134 if (config_.GetPublicKeyCertificateInfo()
135 .public_key()
136 .public_key_info()
137 .type() == CertificateKeyInfo::REPLICA) {
138 hb_info.set_primary(GetPrimary());
139 hb_info.set_version(GetVersion());
140 }
141 LOG(ERROR) << " server:" << config_.GetSelfInfo().id() << " sends HB"
142 << " is ready:" << is_ready_
143 << " client size:" << client_replicas.size()
144 << " svr size:" << replicas.size();
145
146 Request request;
147 request.set_type(Request::TYPE_HEART_BEAT);
148 request.mutable_region_info()->set_region_id(
149 config_.GetConfigData().self_region_id());
150 hb_info.SerializeToString(request.mutable_data());
151
152 int ret = client->SendHeartBeat(request);
153 if (ret <= 0) {
154 LOG(ERROR) << " server:" << config_.GetSelfInfo().id()
155 << " sends HB fail:" << ret;
156 }
157}

159// Porcess the packages received from the network.
160// context contains the client socket which can be used for sending response to
161// the client, the signature for the request will be filled inside the context
162// when parsing the message.
163int ConsensusManager::Process(std::unique_ptr<Context> context,
164 std::unique_ptr<DataInfo> request_info) {
165 global_stats_->IncClientCall();
166 // Decode the whole message, it includes the certificate and data.
167 ResDBMessage message;
168 if (!message.ParseFromArray(request_info->buff, request_info->data_len)) {
169 LOG(ERROR) << "parse data info fail";
170 return -1;
171 }
172
173 std::unique_ptr<Request> request = std::make_unique<Request>();
174 if (!request->ParseFromString(message.data())) {
175 LOG(ERROR) << "parse data info fail";
176 return -1;
177 }
178
179 if (request->type() == Request::TYPE_HEART_BEAT) {
180 return Dispatch(std::move(context), std::move(request));
181 }
182
183 // Check if the certificate is valid.
184 if (message.has_signature() && verifier_) {
185 bool valid = verifier_->VerifyMessage(message.data(), message.signature());
186 if (!valid) {
187 LOG(ERROR) << "request is not valid:"
188 << message.signature().DebugString();
189 LOG(ERROR) << " msg:" << message.data().size()
190 << " is recovery:" << request->is_recovery();
191 return -2;
192 }
193 } else {
194 }
195
196 // forward the signature to the request so that it can be included in the
197 // request/response set if needed.
198 context->signature = message.signature();
199 // LOG(ERROR) << "======= server:" << config_.GetSelfInfo().id()
200 // << " get request type:" << request->type()
201 // << " from:" << request->sender_id();
202
203 return Dispatch(std::move(context), std::move(request));
204}

206// Dispatch the request if it is a heart beat message from other replica or a
207// cert notification from clients. Otherwise, forward to the worker.
208int ConsensusManager::Dispatch(std::unique_ptr<Context> context,
209 std::unique_ptr<Request> request) {
210 if (request->type() == Request::TYPE_HEART_BEAT) {
211 return ProcessHeartBeat(std::move(context), std::move(request));
212 }
213 return ConsensusCommit(std::move(context), std::move(request));
214}

216int ConsensusManager::ProcessHeartBeat(std::unique_ptr<Context> context,
217 std::unique_ptr<Request> request) {
218 std::unique_lock<std::mutex> lk(hb_mutex_);
219 std::vector<ReplicaInfo> replicas = GetReplicas();
220 HeartBeatInfo hb_info;
221 if (!hb_info.ParseFromString(request->data())) {
222 LOG(ERROR) << "parse replica info fail\n";
223 return -1;
224 }
225
226 LOG(ERROR) << "receive public size:" << hb_info.public_keys().size()
227 << " primary:" << hb_info.primary()
228 << " version:" << hb_info.version()
229 << " from region:" << request->region_info().region_id()
230 << " sender:" << hb_info.sender()
231 << " last send:" << hb_info.hb_version()
232 << " current v:" << hb_[hb_info.sender()];
233
234 if (request->region_info().region_id() ==
235 config_.GetConfigData().self_region_id()) {
236 if (config_.GetPublicKeyCertificateInfo()
237 .public_key()
238 .public_key_info()
239 .type() == CertificateKeyInfo::CLIENT) {
240 // TODO count 2f+1 before setting a new primary
241 SetPrimary(hb_info.primary(), hb_info.version());
242 }
243 }
244
245 int replica_num = 0;
246 // Update the public keys received from others.
247 for (const auto& public_key : hb_info.public_keys()) {
248 if (verifier_ && !verifier_->AddPublicKey(public_key)) {
249 LOG(ERROR) << "set public key fail from:"
250 << public_key.public_key_info().node_id();
251 continue;
252 }
253 if (request->region_info().region_id() !=
254 config_.GetConfigData().self_region_id()) {
255 // LOG(ERROR) << "key from other region:"
256 // << request->region_info().region_id();
257 continue;
258 }
259
260 ReplicaInfo info;
261 info.set_ip(public_key.public_key_info().ip());
262 info.set_port(public_key.public_key_info().port());
263 info.set_id(public_key.public_key_info().node_id());
264 if (info.ip().empty()) {
265 LOG(ERROR) << "public doesn't have ip, skip";
266 continue;
267 }
268 // Check whether there is a new replica joining.
269 // TODO notify new replica
270 if (public_key.public_key_info().type() == CertificateKeyInfo::REPLICA) {
271 replica_num++;
272 if (!ReplicaExisted(info, replicas)) {
273 // AddNewReplica(info);
274 }
275 } else {
276 if (!ReplicaExisted(info, clients_)) {
277 AddNewClient(info);
278 }
279 }
280 }
281
282 if (!hb_info.ip().empty() && hb_info.hb_version() > 0 &&
283 hb_[hb_info.sender()] != hb_info.hb_version()) {
284 ReplicaInfo info;
285 info.set_ip(hb_info.ip());
286 info.set_port(hb_info.port());
287 info.set_id(hb_info.sender());
288 // bc_client_->Flush(info);
289 hb_[hb_info.sender()] = hb_info.hb_version();
290 SendHeartBeat();
291 }
292
293 if (!is_ready_ && replica_num >= config_.GetMinDataReceiveNum()) {
294 LOG(ERROR) << "============ Server " << config_.GetSelfInfo().id()
295 << " is ready "
296 "=====================";
297 is_ready_ = true;
298 }
299 return 0;
300}

329void ConsensusManager::SendMessage(const google::protobuf::Message& message,
330 int64_t node_id) {
331 std::vector<ReplicaInfo> replicas = GetReplicas();
332 ReplicaInfo target_replica;
333 for (const auto& replica : replicas) {
334 if (replica.id() == node_id) {
335 target_replica = replica;
336 break;
337 }
338 }
339
340 if (target_replica.ip().empty()) {
341 return;
342 }
343
344 int ret = bc_client_->SendMessage(message, target_replica);
345 if (ret < 0) {
346 LOG(ERROR) << "broadcast request fail:";
347 }
348}
