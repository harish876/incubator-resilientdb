70class Input(object):
71 """! A Input is used to spend assets locked by an Output.
72 Wraps around a Crypto-condition Fulfillment.
73
74 fulfillment (:class:`cryptoconditions.Fulfillment`): A Fulfillment
75 to be signed with a private key.
76 owners_before (:obj:`list` of :obj:`str`): A list of owners after a
77 Transaction was confirmed.
78 fulfills (:class:`~resdb.transaction. TransactionLink`,
79 optional): A link representing the input of a `TRANSFER`
80 Transaction.
81 """
82
83 def __init__(self, fulfillment, owners_before, fulfills=None):
84 """! Create an instance of an :class:`~.Input`.
85 @param fulfillment (:class:`cryptoconditions.Fulfillment`): A
86 Fulfillment to be signed with a private key.
87 @param owners_before (:obj:`list` of :obj:`str`): A list of owners
88 after a Transaction was confirmed.
89 @param fulfills (:class:`~resdb.transaction.
90 TransactionLink`, optional): A link representing the input
91 of a `TRANSFER` Transaction.
92 """
93 if fulfills is not None and not isinstance(fulfills, TransactionLink):
94 raise TypeError("`fulfills` must be a TransactionLink instance")
95 if not isinstance(owners_before, list):
96 raise TypeError("`owners_after` must be a list instance")
97
98 self.fulfillment = fulfillment
99 self.fulfills = fulfills
100 self.owners_before = owners_before
101
102 def __eq__(self, other):
103 # TODO: If `other !== Fulfillment` return `False`
104 return self.to_dict() == other.to_dict()
105
106 def to_dict(self):
107 """! Transforms the object to a Python dictionary.
108 If an Input hasn't been signed yet, this method returns a
109 dictionary representation.
110
111 @return dict: The Input as an alternative serialization format.
112 """
113 try:
114 fulfillment = self.fulfillment.serialize_uri()
115 except (TypeError, AttributeError, ASN1EncodeError, ASN1DecodeError):
116 fulfillment = _fulfillment_to_details(self.fulfillment)
117
118 try:
119 # NOTE: `self.fulfills` can be `None` and that's fine
120 fulfills = self.fulfills.to_dict()
121 except AttributeError:
122 fulfills = None
123
124 input_ = {
125 "owners_before": self.owners_before,
126 "fulfills": fulfills,
127 "fulfillment": fulfillment,
128 }
129 return input_

166def _fulfillment_to_details(fulfillment):
167 """! Encode a fulfillment as a details dictionary
168 Args:
169 @param fulfillment (:class:`cryptoconditions.Fulfillment`): Crypto-conditions Fulfillment object
170 """
171
172 if fulfillment.type_name == "ed25519-sha-256":
173 return {
174 "type": "ed25519-sha-256",
175 "public_key": base58.b58encode(fulfillment.public_key).decode(),
176 }
177
178 if fulfillment.type_name == "threshold-sha-256":
179 subconditions = [
180 _fulfillment_to_details(cond["body"]) for cond in fulfillment.subconditions
181 ]
182 return {
183 "type": "threshold-sha-256",
184 "threshold": fulfillment.threshold,
185 "subconditions": subconditions,
186 }
187
188 raise UnsupportedTypeError(fulfillment.type_name)
189
190
191def _fulfillment_from_details(data, _depth=0):
192 """! Load a fulfillment for a signing spec dictionary
193 @param data tx.output[].condition.details dictionary
194 """
195 if _depth == 100:
196 raise ThresholdTooDeep()
197
198 if data["type"] == "ed25519-sha-256":
199 public_key = base58.b58decode(data["public_key"])
200 return Ed25519Sha256(public_key=public_key)
201
202 if data["type"] == "threshold-sha-256":
203 threshold = ThresholdSha256(data["threshold"])
204 for cond in data["subconditions"]:
205 cond = _fulfillment_from_details(cond, _depth + 1)
206 threshold.add_subfulfillment(cond)
207 return threshold
208
209 raise UnsupportedTypeError(data.get("type"))

212class TransactionLink(object):
213 """! An object for unidirectional linking to a Transaction's Output.
214 Attributes:
215 txid (str, optional): A Transaction to link to.
216 output (int, optional): An output's index in a Transaction with id
217 `txid`.
218 """
219
220 def __init__(self, txid=None, output=None):
221 """! Create an instance of a :class:`~.TransactionLink`.
222 Note:
223 In an IPLD implementation, this class is not necessary anymore,
224 as an IPLD link can simply point to an object, as well as an
225 objects properties. So instead of having a (de)serializable
226 class, we can have a simple IPLD link of the form:
227 `/<tx_id>/transaction/outputs/<output>/`.
228
229 @param txid (str): A Transaction to link to.
230 @param output An (int): Outputs's index in a Transaction
231 with id `txid`.
232 """
233 self.txid = txid
234 self.output = output
235
236 def __bool__(self):
237 return self.txid is not None and self.output is not None
238
239 def __eq__(self, other):
240 # TODO: If `other !== TransactionLink` return `False`
241 return self.to_dict() == other.to_dict()
242
243 def __hash__(self):
244 return hash((self.txid, self.output))
245
246 @classmethod
247 def from_dict(cls, link):
248 """! Transforms a Python dictionary to a TransactionLink object.
249
250 @param link (dict): The link to be transformed.
251
252 @return :class:`~resdb.transaction.TransactionLink`
253 """
254 try:
255 return cls(link["transaction_id"], link["output_index"])
256 except TypeError:
257 return cls()
258
259 def to_dict(self):
260 """! Transforms the object to a Python dictionary.
261 @return The link as an alternative serialization format.
262 """
263 if self.txid is None and self.output is None:
264 return None
265 else:
266 return {
267 "transaction_id": self.txid,
268 "output_index": self.output,
269 }
270
271 def to_uri(self, path=""):
272 if self.txid is None and self.output is None:
273 return None
274 return "{}/transactions/{}/outputs/{}".format(path, self.txid, self.output)

277class Output(object):
278 """! An Output is used to lock an asset.
279 Wraps around a Crypto-condition Condition.
280 Attributes:
281 fulfillment (:class:`cryptoconditions.Fulfillment`): A Fulfillment
282 to extract a Condition from.
283 public_keys (:obj:`list` of :obj:`str`, optional): A list of
284 owners before a Transaction was confirmed.
285 """
286
287 MAX_AMOUNT = 9 * 10**18
288
289 def __init__(self, fulfillment, public_keys=None, amount=1):
290 """! Create an instance of a :class:`~.Output`.
291 Args:
292 @param fulfillment (:class:`cryptoconditions.Fulfillment`): A
293 Fulfillment to extract a Condition from.
294 @param public_keys (:obj:`list` of :obj:`str`, optional): A list of
295 owners before a Transaction was confirmed.
296 @param amount (int): The amount of Assets to be locked with this
297 Output.
298
299 @exception TypeError: if `public_keys` is not instance of `list`.
300 """
301 if not isinstance(public_keys, list) and public_keys is not None:
302 raise TypeError("`public_keys` must be a list instance or None")
303 if not isinstance(amount, int):
304 raise TypeError("`amount` must be an int")
305 if amount < 1:
306 raise AmountError("`amount` must be greater than 0")
307 if amount > self.MAX_AMOUNT:
308 raise AmountError("`amount` must be <= %s" % self.MAX_AMOUNT)
309
310 self.fulfillment = fulfillment
311 self.amount = amount
312 self.public_keys = public_keys
313
314 def __eq__(self, other):
315 # TODO: If `other !== Condition` return `False`
316 return self.to_dict() == other.to_dict()
317
318 def to_dict(self):
319 """! Transforms the object to a Python dictionary.
320 Note:
321 A dictionary serialization of the Input the Output was
322 derived from is always provided.
323
324 @return The Output as an alternative serialization format.
325 """
326 # TODO FOR CC: It must be able to recognize a hashlock condition
327 # and fulfillment!
328 condition = {}
329 try:
330 condition["details"] = _fulfillment_to_details(self.fulfillment)
331 except AttributeError:
332 pass
333
334 try:
335 condition["uri"] = self.fulfillment.condition_uri
336 except AttributeError:
337 condition["uri"] = self.fulfillment
338
339 output = {
340 "public_keys": self.public_keys,
341 "condition": condition,
342 "amount": str(self.amount),
343 }
344 return output

346 @classmethod
347 def generate(cls, public_keys, amount):
348 """! Generates a Output from a specifically formed tuple or list.
349 Note:
350 If a ThresholdCondition has to be generated where the threshold
351 is always the number of subconditions it is split between, a
352 list of the following structure is sufficient:
353 [(address|condition)*, [(address|condition)*, ...], ...]
354
355 @param public_keys (:obj:`list` of :obj:`str`): The public key of
356 the users that should be able to fulfill the Condition
357 that is being created.
358 @param amount (:obj:`int`): The amount locked by the Output.
359 @return An Output that can be used in a Transaction.
360
361 @exception TypeError: If `public_keys` is not an instance of `list`.
362 @exception ValueError: If `public_keys` is an empty list.
363 """
364 threshold = len(public_keys)
365 if not isinstance(amount, int):
366 raise TypeError("`amount` must be a int")
367 if amount < 1:
368 raise AmountError("`amount` needs to be greater than zero")
369 if not isinstance(public_keys, list):
370 raise TypeError("`public_keys` must be an instance of list")
371 if len(public_keys) == 0:
372 raise ValueError(
373 "`public_keys` needs to contain at least one" "owner")
374 elif len(public_keys) == 1 and not isinstance(public_keys[0], list):
375 if isinstance(public_keys[0], Fulfillment):
376 ffill = public_keys[0]
377 else:
378 ffill = Ed25519Sha256(
379 public_key=base58.b58decode(public_keys[0]))
380 return cls(ffill, public_keys, amount=amount)
381 else:
382 # Threshold conditions not supported by resdb yet
383 initial_cond = ThresholdSha256(threshold=threshold)
384 threshold_cond = reduce(
385 cls._gen_condition, public_keys, initial_cond)
386 return cls(threshold_cond, public_keys, amount=amount)
387
388 @classmethod
389 def _gen_condition(cls, initial, new_public_keys):
390 """! Generates ThresholdSha256 conditions from a list of new owners.
391 Note:
392 This method is intended only to be used with a reduce function.
393 For a description on how to use this method, see
394 :meth:`~.Output.generate`.
395 Args:
396 @param initial (:class:`cryptoconditions.ThresholdSha256`): A Condition representing the overall root.
397 @param new_public_keys (:obj:`list` of :obj:`str`|str): A list of new
398 owners or a single new owner.
399 @return :class:`cryptoconditions.ThresholdSha256`:
400 """
401 try:
402 threshold = len(new_public_keys)
403 except TypeError:
404 threshold = None
405
406 if isinstance(new_public_keys, list) and len(new_public_keys) > 1:
407 ffill = ThresholdSha256(threshold=threshold)
408 reduce(cls._gen_condition, new_public_keys, ffill)
409 elif isinstance(new_public_keys, list) and len(new_public_keys) <= 1:
410 raise ValueError("Sublist cannot contain single owner")
411 else:
412 try:
413 new_public_keys = new_public_keys.pop()
414 except AttributeError:
415 pass
416 # NOTE: Instead of submitting base58 encoded addresses, a user
417 # of this class can also submit fully instantiated
418 # Cryptoconditions. In the case of casting
419 # `new_public_keys` to a Ed25519Fulfillment with the
420 # result of a `TypeError`, we're assuming that
421 # `new_public_keys` is a Cryptocondition then.
422 if isinstance(new_public_keys, Fulfillment):
423 ffill = new_public_keys
424 else:
425 ffill = Ed25519Sha256(
426 public_key=base58.b58decode(new_public_keys))
427 initial.add_subfulfillment(ffill)
428 return initial

430 @classmethod
431 def from_dict(cls, data):
432 """! Transforms a Python dictionary to an Output object.
433 Note:
434 To pass a serialization cycle multiple times, a
435 Cryptoconditions Fulfillment needs to be present in the
436 passed-in dictionary, as Condition URIs are not serializable
437 anymore.
438
439 @param data (dict): The dict to be transformed.
440 @return :class:`~resdb.transaction.Output`
441 """
442 try:
443 fulfillment = _fulfillment_from_details(
444 data["condition"]["details"])
445 except KeyError:
446 # NOTE: Hashlock condition case
447 fulfillment = data["condition"]["uri"]
448 try:
449 amount = int(data["amount"])
450 except ValueError:
451 raise AmountError("Invalid amount: %s" % data["amount"])
452 return cls(fulfillment, data["public_keys"], amount)

455class Transaction(object):
456 """! A Transaction is used to create and transfer assets.
457 Note:
458 For adding Inputs and Outputs, this class provides methods
459 to do so.
460 Attributes:
461 operation (str): Defines the operation of the Transaction.
462 inputs (:obj:`list` of :class:`~resdb.
463 transaction.Input`, optional): Define the assets to
464 spend.
465 outputs (:obj:`list` of :class:`~resdb.
466 transaction.Output`, optional): Define the assets to lock.
467 asset (dict): Asset payload for this Transaction. ``CREATE``
468 Transactions require a dict with a ``data``
469 property while ``TRANSFER`` Transactions require a dict with a
470 ``id`` property.
471 metadata (dict):
472 Metadata to be stored along with the Transaction.
473 version (string): Defines the version number of a Transaction.
474 """
475
476 CREATE = "CREATE"
477 TRANSFER = "TRANSFER"
478 ALLOWED_OPERATIONS = (CREATE, TRANSFER)
479 VERSION = "2.0"
480
481 def __init__(
482 self,
483 operation,
484 asset,
485 inputs=None,
486 outputs=None,
487 metadata=None,
488 version=None,
489 hash_id=None,
490 ):
491 """! The constructor allows to create a customizable Transaction.
492 Note:
493 When no `version` is provided, one is being
494 generated by this method.
495
496 @param operation (str): Defines the operation of the Transaction.
497 @param asset (dict): Asset payload for this Transaction.
498 @param inputs (:obj:`list` of :class:`~resdb.transaction.Input`, optional):Define the assets to
499 @param outputs (:obj:`list` of :class:`~resdb.transaction.Output`, optional):Define the assets to lock.
500 @param metadata (dict): Metadata to be stored along with the Transaction.
501 @param version (string): Defines the version number of a Transaction.
502 @param hash_id (string): Hash id of the transaction.
503 """
504 if operation not in Transaction.ALLOWED_OPERATIONS:
505 allowed_ops = ", ".join(self.__class__.ALLOWED_OPERATIONS)
506 raise ValueError(
507 "`operation` must be one of {}".format(allowed_ops))
508
509 # Asset payloads for 'CREATE' operations must be None or
510 # dicts holding a `data` property. Asset payloads for 'TRANSFER'
511 # operations must be dicts holding an `id` property.
512 if (
513 operation == Transaction.CREATE
514 and asset is not None
515 and not (isinstance(asset, dict) and "data" in asset)
516 ):
517 raise TypeError(
518 (
519 "`asset` must be None or a dict holding a `data` "
520 " property instance for '{}' "
521 "Transactions".format(operation)
522 )
523 )
524 elif operation == Transaction.TRANSFER and not (
525 isinstance(asset, dict) and "id" in asset
526 ):
527 raise TypeError(
528 (
529 "`asset` must be a dict holding an `id` property "
530 "for 'TRANSFER' Transactions".format(operation)
531 )
532 )
533
534 if outputs and not isinstance(outputs, list):
535 raise TypeError("`outputs` must be a list instance or None")
536
537 if inputs and not isinstance(inputs, list):
538 raise TypeError("`inputs` must be a list instance or None")
539
540 if metadata is not None and not isinstance(metadata, dict):
541 raise TypeError("`metadata` must be a dict or None")
542
543 self.version = version if version is not None else self.VERSION
544 self.operation = operation
545 self.asset = asset
546 self.inputs = inputs or []
547 self.outputs = outputs or []
548 self.metadata = metadata
549 self._id = hash_id

587 @classmethod
588 def create(cls, tx_signers, recipients, metadata=None, asset=None):
589 """! A simple way to generate a `CREATE` transaction.
590 Note:
591 This method currently supports the following Cryptoconditions
592 use cases:
593 - Ed25519
594 - ThresholdSha256
595 Additionally, it provides support for the following Resdb
596 use cases:
597 - Multiple inputs and outputs.
598
599 @param tx_signers (:obj:`list` of :obj:`str`): A list of keys that
600 represent the signers of the CREATE Transaction.
601 @param recipients (:obj:`list` of :obj:`tuple`): A list of
602 ([keys],amount) that represent the recipients of this
603 Transaction.
604 @param metadata (dict): The metadata to be stored along with the
605 Transaction.
606 @param asset (dict): The metadata associated with the asset that will
607 be created in this Transaction.
608
609 @return :class:`~resdb.transaction.Transaction`
610 """
611 if not isinstance(tx_signers, list):
612 raise TypeError("`tx_signers` must be a list instance")
613 if not isinstance(recipients, list):
614 raise TypeError("`recipients` must be a list instance")
615 if len(tx_signers) == 0:
616 raise ValueError("`tx_signers` list cannot be empty")
617 if len(recipients) == 0:
618 raise ValueError("`recipients` list cannot be empty")
619 if not (asset is None or isinstance(asset, dict)):
620 raise TypeError("`asset` must be a dict or None")
621
622 inputs = []
623 outputs = []
624
625 # generate_outputs
626 for recipient in recipients:
627 if not isinstance(recipient, tuple) or len(recipient) != 2:
628 raise ValueError(
629 (
630 "Each `recipient` in the list must be a"
631 " tuple of `([<list of public keys>],"
632 " <amount>)`"
633 )
634 )
635 pub_keys, amount = recipient
636 outputs.append(Output.generate(pub_keys, amount))
637
638 # generate inputs
639 inputs.append(Input.generate(tx_signers))
640
641 return cls(cls.CREATE, {"data": asset}, inputs, outputs, metadata)

643 @classmethod
644 def transfer(cls, inputs, recipients, asset_id, metadata=None):
645 """! A simple way to generate a `TRANSFER` transaction.
646 Note:
647 Different cases for threshold conditions:
648 Combining multiple `inputs` with an arbitrary number of
649 `recipients` can yield interesting cases for the creation of
650 threshold conditions we'd like to support. The following
651 notation is proposed:
652 1. The index of a `recipient` corresponds to the index of
653 an input:
654 e.g. `transfer([input1], [a])`, means `input1` would now be
655 owned by user `a`.
656 2. `recipients` can (almost) get arbitrary deeply nested,
657 creating various complex threshold conditions:
658 e.g. `transfer([inp1, inp2], [[a, [b, c]], d])`, means
659 `a`'s signature would have a 50% weight on `inp1`
660 compared to `b` and `c` that share 25% of the leftover
661 weight respectively. `inp2` is owned completely by `d`.
662
663 @param inputs (:obj:`list` of :class:`~resdb.transaction.Input`): Converted `Output`s, intended to
664 be used as inputs in the transfer to generate.
665 @param recipients (:obj:`list` of :obj:`tuple`): A list of
666 ([keys],amount) that represent the recipients of this
667 Transaction.
668 @param asset_id (str): The asset ID of the asset to be transferred in
669 this Transaction.
670 @param metadata (dict): Python dictionary to be stored along with the
671 Transaction.
672
673 @return :class:`~resdb.transaction.Transaction`
674 """
675 if not isinstance(inputs, list):
676 raise TypeError("`inputs` must be a list instance")
677 if len(inputs) == 0:
678 raise ValueError("`inputs` must contain at least one item")
679 if not isinstance(recipients, list):
680 raise TypeError("`recipients` must be a list instance")
681 if len(recipients) == 0:
682 raise ValueError("`recipients` list cannot be empty")
683
684 outputs = []
685 for recipient in recipients:
686 if not isinstance(recipient, tuple) or len(recipient) != 2:
687 raise ValueError(
688 (
689 "Each `recipient` in the list must be a"
690 " tuple of `([<list of public keys>],"
691 " <amount>)`"
692 )
693 )
694 pub_keys, amount = recipient
695 outputs.append(Output.generate(pub_keys, amount))
696
697 if not isinstance(asset_id, str):
698 raise TypeError("`asset_id` must be a string")
699
700 inputs = deepcopy(inputs)
701 return cls(cls.TRANSFER, {"id": asset_id}, inputs, outputs, metadata)

710 def to_inputs(self, indices=None):
711 """! Converts a Transaction's outputs to spendable inputs.
712 Note:
713 Takes the Transaction's outputs and derives inputs
