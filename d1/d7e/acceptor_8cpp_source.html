48void Acceptor::Run() {
49 LOG(ERROR) << "server:" << config_.GetSelfInfo().id() << " start running";
50
51 LockFreeQueue<Socket> socket_queue("server");
52 std::vector<std::thread> threads;
53 int woker_num = config_.GetInputWorkerNum();
54 for (int i = 0; i < woker_num; ++i) {
55 threads.push_back(std::thread([&]() {
56 while (IsRunning()) {
57 auto client_socket = socket_queue.Pop();
58 if (client_socket == nullptr) {
59 continue;
60 }
61 std::unique_ptr<DataInfo> request_info = std::make_unique<DataInfo>();
62 int ret =
63 client_socket->Recv(&request_info->buff, &request_info->data_len);
64 if (ret <= 0) {
65 continue;
66 }
67 std::unique_ptr<QueueItem> item = std::make_unique<QueueItem>();
68 item->socket = std::move(client_socket);
69 item->data = std::move(request_info);
70 global_stats_->ServerCall();
71 input_queue_->Push(std::move(item));
72 }
73 }));
74 }
75
76 while (IsRunning()) {
77 auto client_socket = socket_->Accept();
78 if (client_socket == nullptr) {
79 continue;
80 }
81 socket_queue.Push(std::move(client_socket));
82 }
83
84 for (auto& th : threads) {
85 if (th.joinable()) {
86 th.join();
87 }
88 }
89}
