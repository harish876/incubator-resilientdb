28void ShiftManager::AddSliceInfo(const SliceInfo& slice_info) {

36bool ShiftManager::Check(const SliceInfo& slice_info, int timeout_ms) {
37 auto check_done = [&]() {
38 size_t n = config_.GetReplicaNum();
39 size_t f = config_.GetMaxMaliciousReplicaNum();
40 return data_[std::make_pair(slice_info.height(), slice_info.shift_idx())]
41 .size() >= n - f;
42 };
43 std::unique_lock<std::mutex> lk(mutex_);
44 cv_.wait_for(lk, std::chrono::microseconds(timeout_ms),
45 [&] { return check_done(); });
46 LOG(ERROR)
47 << "get shift msg:" << slice_info.height() << " "
48 << slice_info.shift_idx() << ":"
49 << data_[std::make_pair(slice_info.height(), slice_info.shift_idx())]
50 .size();
51 return check_done();
52}
