63 def post_transaction(self, transaction, mode):
64 """Submit a valid transaction to the mempool."""
65 if not mode or mode not in self.mode_list:
66 raise ValidationError('Mode must be one of the following {}.'
67 .format(', '.join(self.mode_list)))
68
69 tx_dict = transaction.tx_dict if transaction.tx_dict else transaction.to_dict()
70 payload = {
71 'method': mode,
72 'jsonrpc': '2.0',
73 'params': [encode_transaction(tx_dict)],
74 'id': str(uuid4())
75 }
76 # TODO: handle connection errors!
77 return requests.post(self.endpoint, json=payload)
78
79 def write_transaction(self, transaction, mode):
80 # This method offers backward compatibility with the Web API.
81 """Submit a valid transaction to the mempool."""
82 response = self.post_transaction(transaction, mode)
83 return self._process_post_response(response.json(), mode)
84
85 def _process_post_response(self, response, mode):
86 logger.debug(response)
87
88 error = response.get('error')
89 if error:
90 status_code = 500
91 message = error.get('message', 'Internal Error')
92 data = error.get('data', '')
93
94 if 'Tx already exists in cache' in data:
95 status_code = 400
96
97 return (status_code, message + ' - ' + data)
98
99 result = response['result']
100 if mode == self.mode_commit:
101 check_tx_code = result.get('check_tx', {}).get('code', 0)
102 deliver_tx_code = result.get('deliver_tx', {}).get('code', 0)
103 error_code = check_tx_code or deliver_tx_code
104 else:
105 error_code = result.get('code', 0)
106
107 if error_code:
108 return (500, 'Transaction validation failed')
109
110 return (202, '')

136 def update_utxoset(self, transaction):
137 """Update the UTXO set given ``transaction``. That is, remove
138 the outputs that the given ``transaction`` spends, and add the
139 outputs that the given ``transaction`` creates.
140
141 Args:
142 transaction (:obj:`~resdb_validator.models.Transaction`): A new
143 transaction incoming into the system for which the UTXO
144 set needs to be updated.
145 """
146 spent_outputs = [
147 spent_output for spent_output in transaction.spent_outputs
148 ]
149 if spent_outputs:
150 self.delete_unspent_outputs(*spent_outputs)
151 self.store_unspent_outputs(
152 *[utxo._asdict() for utxo in transaction.unspent_outputs]
153 )

166 def get_utxoset_merkle_root(self):
167 """Returns the merkle root of the utxoset. This implies that
168 the utxoset is first put into a merkle tree.
169
170 For now, the merkle tree and its root will be computed each
171 time. This obviously is not efficient and a better approach
172 that limits the repetition of the same computation when
173 unnecesary should be sought. For instance, future optimizations
174 could simply re-compute the branches of the tree that were
175 affected by a change.
176
177 The transaction hash (id) and output index should be sufficient
178 to uniquely identify a utxo, and consequently only that
179 information from a utxo record is needed to compute the merkle
180 root. Hence, each node of the merkle tree should contain the
181 tuple (txid, output_index).
182
183 .. important:: The leaves of the tree will need to be sorted in
184 some kind of lexicographical order.
185
186 Returns:
187 str: Merkle root in hexadecimal form.
188 """
189 utxoset = backend.query.get_unspent_outputs(self.connection)
190 # TODO Once ready, use the already pre-computed utxo_hash field.
191 # See common/transactions.py for details.
192 hashes = [
193 sha3_256(
194 '{}{}'.format(utxo['transaction_id'], utxo['output_index']).encode()
195 ).digest() for utxo in utxoset
196 ]
197 # TODO Notice the sorted call!
198 return merkleroot(sorted(hashes))

276 def get_spent(self, txid, output, current_transactions=[]):
277 transactions = backend.query.get_spent(self.connection, txid,
278 output)
279 transactions = list(transactions) if transactions else []
280 if len(transactions) > 1:
281 raise core_exceptions.CriticalDoubleSpend(
282 '`{}` was spent more than once. There is a problem'
283 ' with the chain'.format(txid))
284
285 current_spent_transactions = []
286 for ctxn in current_transactions:
287 for ctxn_input in ctxn.inputs:
288 if ctxn_input.fulfills and\
289 ctxn_input.fulfills.txid == txid and\
290 ctxn_input.fulfills.output == output:
291 current_spent_transactions.append(ctxn)
292
293 transaction = None
294 if len(transactions) + len(current_spent_transactions) > 1:
295 raise DoubleSpend('tx "{}" spends inputs twice'.format(txid))
296 elif transactions:
297 transaction = Transaction.from_db(self, transactions[0])
298 elif current_spent_transactions:
299 transaction = current_spent_transactions[0]
300
301 return transaction

313 def get_block(self, block_id):
314 """Get the block with the specified `block_id`.
315
316 Returns the block corresponding to `block_id` or None if no match is
317 found.
318
319 Args:
320 block_id (int): block id of the block to get.
321 """
322
323 block = backend.query.get_block(self.connection, block_id)
324 latest_block = self.get_latest_block()
325 latest_block_height = latest_block['height'] if latest_block else 0
326
327 if not block and block_id > latest_block_height:
328 return
329
330 result = {'height': block_id,
331 'transactions': []}
332
333 if block:
334 transactions = backend.query.get_transactions(self.connection, block['transactions'])
335 result['transactions'] = [t.to_dict() for t in Transaction.from_db(self, transactions)]
336
337 return result

355 def validate_transaction(self, tx, current_transactions=[]):
356 """Validate a transaction against the current status of the database."""
357
358 transaction = tx
359
360 # CLEANUP: The conditional below checks for transaction in dict format.
361 # It would be better to only have a single format for the transaction
362 # throught the code base.
363 if isinstance(transaction, dict):
364 try:
365 transaction = Transaction.from_dict(tx)
366 except SchemaValidationError as e:
367 logger.warning('Invalid transaction schema: %s', e.__cause__.message)
368 return False
369 except ValidationError as e:
370 logger.warning('Invalid transaction (%s): %s', type(e).__name__, e)
371 return False
372 return transaction.validate(self, current_transactions)

461 def migrate_abci_chain(self):
462 """Generate and record a new ABCI chain ID. New blocks are not
463 accepted until we receive an InitChain ABCI request with
464 the matching chain ID and validator set.
465
466 Chain ID is generated based on the current chain and height.
467 `chain-X` => `chain-X-migrated-at-height-5`.
468 `chain-X-migrated-at-height-5` => `chain-X-migrated-at-height-21`.
469
470 If there is no known chain (we are at genesis), the function returns.
471 """
472 latest_chain = self.get_latest_abci_chain()
473 if latest_chain is None:
474 return
475
476 block = self.get_latest_block()
477
478 suffix = '-migrated-at-height-'
479 chain_id = latest_chain['chain_id']
480 block_height_str = str(block['height'])
481 new_chain_id = chain_id.split(suffix)[0] + suffix + block_height_str
482
483 self.store_abci_chain(block['height'] + 1, new_chain_id, False)
