60void TransactionAccessor::TransactionFetching() {
61 std::unique_ptr<ResDBTxnAccessor> client = GetResDBTxnAccessor();
62 assert(client != nullptr);
63
64 uint64_t last_time = 0;
65 uint64_t current_time = GetCurrentTime();
66 sleep(10);
67 int cur_seq_pbft = 1;
68 srand(12345);
69
70 while (!stop_) {
71 uint64_t cur_seq = max_received_seq_ + 1;
72 if (cur_seq > 10000000) break;
73 auto ret = client->GetTxn(cur_seq, cur_seq + 1000);
74 if (!ret.ok() || (*ret).size() == 0 || (*ret)[0].first != cur_seq) {
75 // LOG(ERROR) << "get txn fail:" << cur_seq;
76 sleep(1);
77 continue;
78 }
79
80 double need_time = 2.0 / 1000;
81 uint64_t current_time1 = GetCurrentTime();
82 for (auto& res : (*ret)) {
83 uint64_t current_time = GetCurrentTime();
84
85 if (last_time > 0) {
86 double sleep_time =
87 std::max((uint64_t)0, current_time - last_time) / 1000000.0;
88 // LOG(ERROR)<<"sleep:"<<sleep_time<<" "<<(need_time-sleep_time)<<"
89 //"<<(int)((need_time - sleep_time)*1000000);
90 if (need_time - sleep_time > 0) {
91 usleep((int)((need_time - sleep_time) * 1000000));
92 }
93 }
94
95 // LOG(ERROR)<<"current time:"<<current_time;
96 last_time = current_time;
97 std::unique_ptr<ClientTransactions> client_txn =
98 std::make_unique<ClientTransactions>();
99 client_txn->set_transaction_data(res.second);
100 client_txn->set_seq(cur_seq);
101 client_txn->set_create_time(GetCurrentTime());
102 // prometheus_handler_->Inc(TRANSACTION_INPUT, 1);
103 queue_.Push(std::move(client_txn));
104 cur_seq++;
105 }
106 max_received_seq_ = cur_seq;
107
108 std::lock_guard<std::mutex> lk(mutex_);
109 cv_.notify_all();
110 }
111 return;
112}

118// obtain [seq, seq+batch_num-1] transactions
119std::unique_ptr<BatchClientTransactions>
120TransactionAccessor::ConsumeTransactions(uint64_t seq) {
121 LOG(ERROR) << "consume transaction:" << seq
122 << " batch:" << config_.BatchTransactionNum()
123 << " received max seq:" << max_received_seq_;
124 if (seq + config_.BatchTransactionNum() > max_received_seq_ + 1) {
125 std::unique_lock<std::mutex> lk(mutex_);
126 cv_.wait_for(lk, std::chrono::seconds(1), [&] {
127 return seq + config_.BatchTransactionNum() <= max_received_seq_ + 1;
128 });
129
130 return nullptr;
131 }
132 while (seq > next_consume_) {
133 *queue_.Pop();
134 next_consume_++;
135 }
136 if (seq != next_consume_) {
137 LOG(ERROR) << "next should consume:" << next_consume_;
138 return nullptr;
139 }
140
141 std::unique_ptr<BatchClientTransactions> batch_transactions =
142 std::make_unique<BatchClientTransactions>();
143 for (uint32_t i = 0; i < config_.BatchTransactionNum(); ++i) {
144 *batch_transactions->add_transactions() = *queue_.Pop();
145 }
146 batch_transactions->set_min_seq(seq);
147 batch_transactions->set_max_seq(seq + config_.BatchTransactionNum() - 1);
148 next_consume_ = next_consume_ + config_.BatchTransactionNum();
149 LOG(ERROR) << "get batch:" << GetCurrentTime();
150 return batch_transactions;
151}
