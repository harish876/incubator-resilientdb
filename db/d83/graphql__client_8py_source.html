43 async def execute_query(self, query: str, variables: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
44 """
45 Execute a GraphQL query.
46
47 Args:
48 query: GraphQL query string
49 variables: Optional variables for the query
50
51 Returns:
52 Response data from GraphQL server
53 """
54 payload = {
55 "query": query,
56 "variables": variables or {}
57 }
58
59 async with httpx.AsyncClient(timeout=self.timeout) as client:
60 response = await client.post(
61 self.url,
62 json=payload,
63 headers=self._get_headers()
64 )
65 response.raise_for_status()
66 result = response.json()
67
68 if "errors" in result:
69 error_msg = json.dumps(result["errors"], indent=2)
70 raise Exception(f"GraphQL errors: {error_msg}")
71
72 return result.get("data", {})

85 async def get_transaction(self, transaction_id: str) -> Dict[str, Any]:
86 """
87 Get transaction by ID via GraphQL.
88 Returns RetrieveTransaction with all available fields.
89
90 Based on official documentation: https://beacon.resilientdb.com/docs/resilientdb_graphql#get-transaction-by-id
91
92 Note: All fields except 'metadata' are NON_NULL and must be included in the query.
93 """
94 query = """
95 query GetTx($id: ID!) {
96 getTransaction(id: $id) {
97 id
98 version
99 amount
100 uri
101 type
102 publicKey
103 operation
104 metadata
105 asset
106 signerPublicKey
107 }
108 }
109 """
110 return await self.execute_query(query, {"id": transaction_id})

112 async def post_transaction(self, data: Dict[str, Any]) -> Dict[str, Any]:
113 """
114 Post a new asset transaction via GraphQL.
115 Requires PrepareAsset with all required fields:
116 - operation (String): Transaction operation type
117 - amount (Int): Transaction amount
118 - signerPublicKey (String): Public key of the signer
119 - signerPrivateKey (String): Private key of the signer
120 - recipientPublicKey (String): Public key of the recipient
121 - asset (JSONScalar): Asset data as JSON object with 'data' field (not string!)
122
123 Based on official documentation: https://beacon.resilientdb.com/docs/resilientdb_graphql#get-transaction-by-id
124
125 The asset must be structured as: {"data": {...}} where {...} contains your actual asset data.
126
127 Returns CommitTransaction with transaction ID.
128 """
129 # Validate required fields
130 required_fields = ["operation", "amount", "signerPublicKey", "signerPrivateKey", "recipientPublicKey", "asset"]
131 missing_fields = [field for field in required_fields if field not in data]
132 if missing_fields:
133 raise Exception(
134 f"Missing required fields in PrepareAsset: {', '.join(missing_fields)}. "
135 f"Required fields: {', '.join(required_fields)}"
136 )
137
138 # IMPORTANT: Keep asset as dict/object - JSONScalar expects JSON object, not string
139 # MCP framework may convert the asset object to a string, so we need to parse it back
140 asset = data["asset"]
141 if isinstance(asset, str):
142 try:
143 asset = json.loads(asset)
144 except json.JSONDecodeError as e:
145 raise Exception(
146 f"Failed to parse asset JSON string: {e}. "
147 f"Asset value: {asset[:100]}..."
148 )
149
150 # Ensure asset is a dict/object (not a list, string, or primitive)
151 if not isinstance(asset, dict):
152 raise Exception(
153 f"Asset must be a JSON object (dict), but got {type(asset).__name__}. "
154 f"Value: {str(asset)[:100]}..."
155 )
156
157 # Update data with parsed asset
158 data["asset"] = asset
159
160 mutation = """
161 mutation Test($data: PrepareAsset!) {
162 postTransaction(data: $data) {
163 id
164 }
165 }
166 """
167 return await self.execute_query(mutation, {"data": data})

180 async def get_key_value(self, key: str) -> Dict[str, Any]:
181 """
182 Query key-value store via HTTP REST API (Crow server on port 18000).
183 """
184 async with httpx.AsyncClient(timeout=self.timeout) as client:
185 response = await client.get(
186 f"{self.http_url}/v1/transactions/{key}",
187 headers=self._get_headers()
188 )
189 response.raise_for_status()
190 result = response.json()
191
192 # Return in a consistent format
193 return {
194 "key": key,
195 "value": result.get("value", result),
196 "response": result
197 }

199 async def set_key_value(self, key: str, value: Any) -> Dict[str, Any]:
200 """
201 Set key-value pair via HTTP REST API (Crow server on port 18000).
202 """
203 # Convert value to string if it's not already
204 if not isinstance(value, str):
205 value = json.dumps(value)
206
207 async with httpx.AsyncClient(timeout=self.timeout) as client:
208 response = await client.post(
209 f"{self.http_url}/v1/transactions/commit",
210 json={"id": key, "value": value},
211 headers=self._get_headers()
212 )
213 response.raise_for_status()
214
215 # Parse response text (Crow returns plain text like "id: key")
216 response_text = response.text.strip()
217
218 return {
219 "key": key,
220 "value": value,
221 "status": "committed",
222 "response": response_text
