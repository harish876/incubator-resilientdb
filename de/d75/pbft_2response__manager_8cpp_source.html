117// =================== response ========================
118// handle the response message. If receive f+1 commit messages, send back to the
119// caller.
120int ResponseManager::ProcessResponseMsg(std::unique_ptr<Context> context,
121 std::unique_ptr<Request> request) {
122 std::unique_ptr<Request> response;
123 // Add the response message, and use the call back to collect the received
124 // messages.
125 // The callback will be triggered if it received f+1 messages.
126 if (request->ret() == -2) {
127 LOG(ERROR) << "get response fail:" << request->ret();
128 send_num_--;
129 return 0;
130 }
131 CollectorResultCode ret =
132 AddResponseMsg(context->signature, std::move(request),
133 [&](const Request& request,
134 const TransactionCollector::CollectorDataType*) {
135 response = std::make_unique<Request>(request);
136 return;
137 });
138
139 if (ret == CollectorResultCode::STATE_CHANGED) {
140 BatchUserResponse batch_response;
141 if (batch_response.ParseFromString(response->data())) {
142 SendResponseToClient(batch_response);
143 } else {
144 LOG(ERROR) << "parse response fail:";
145 }
146 }
147 return ret == CollectorResultCode::INVALID ? -2 : 0;
148}

150bool ResponseManager::MayConsensusChangeStatus(
151 int type, int received_count, std::atomic<TransactionStatue>* status) {
152 switch (type) {
153 case Request::TYPE_RESPONSE:
154 // if receive f+1 response results, ack to the caller.
155 if (*status == TransactionStatue::None &&
156 config_.GetMinClientReceiveNum() <= received_count) {
157 TransactionStatue old_status = TransactionStatue::None;
158 return status->compare_exchange_strong(
159 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
160 std::memory_order_acq_rel);
161 }
162 break;
163 }
164 return false;
165}

167CollectorResultCode ResponseManager::AddResponseMsg(
168 const SignatureInfo& signature, std::unique_ptr<Request> request,
169 std::function<void(const Request&,
170 const TransactionCollector::CollectorDataType*)>
171 response_call_back) {
172 if (request == nullptr) {
173 return CollectorResultCode::INVALID;
174 }
175
176 std::string hash = request->hash();
177
178 std::unique_ptr<BatchUserResponse> batch_response =
179 std::make_unique<BatchUserResponse>();
180 if (!batch_response->ParseFromString(request->data())) {
181 LOG(ERROR) << "parse response fail:" << request->data().size()
182 << " seq:" << request->seq();
183 RemoveWaitingResponseRequest(hash);
184 return CollectorResultCode::INVALID;
185 }
186
187 uint64_t seq = batch_response->local_id();
188 request->set_seq(seq);
189
190 int type = request->type();
191 int resp_received_count = 0;
192 int ret = collector_pool_->GetCollector(seq)->AddRequest(
193 std::move(request), signature, false,
194 [&](const Request& request, int received_count,
195 TransactionCollector::CollectorDataType* data,
196 std::atomic<TransactionStatue>* status, bool force) {
197 if (MayConsensusChangeStatus(type, received_count, status)) {
198 resp_received_count = 1;
199 response_call_back(request, data);
200 }
201 });
202 if (ret != 0) {
203 return CollectorResultCode::INVALID;
204 }
205 if (resp_received_count > 0) {
206 collector_pool_->Update(seq);
207 RemoveWaitingResponseRequest(hash);
208 return CollectorResultCode::STATE_CHANGED;
209 }
210 return CollectorResultCode::OK;
211}

213void ResponseManager::SendResponseToClient(
214 const BatchUserResponse& batch_response) {
215 uint64_t create_time = batch_response.createtime();
216 uint64_t local_id = batch_response.local_id();
217 if (create_time > 0) {
218 uint64_t run_time = GetCurrentTime() - create_time;
219 global_stats_->AddLatency(run_time);
220 } else {
221 LOG(ERROR) << "seq:" << local_id << " no resp";
222 }
223 send_num_--;
224
225 if (config_.IsPerformanceRunning()) {
226 return;
227 }
228
229 std::vector<std::unique_ptr<Context>> context_list =
230 FetchContextList(batch_response.local_id());
231 if (context_list.empty()) {
232 LOG(ERROR) << "context list is empty. local id:"
233 << batch_response.local_id();
234 return;
235 }
236
237 for (size_t i = 0; i < context_list.size(); ++i) {
238 auto& context = context_list[i];
239 if (context->client == nullptr) {
240 LOG(ERROR) << " no channel:";
241 continue;
242 }
243 int ret = context->client->SendRawMessageData(batch_response.response(i));
244 if (ret) {
245 LOG(ERROR) << "send resp fail ret:" << ret;
246 }
247 }
248}

250// =================== request ========================
251int ResponseManager::BatchProposeMsg() {
252 LOG(INFO) << "batch wait time:" << config_.ClientBatchWaitTimeMS()
253 << " batch num:" << config_.ClientBatchNum();
254 std::vector<std::unique_ptr<QueueItem>> batch_req;
255 while (!stop_) {
256 if (send_num_ > config_.GetMaxProcessTxn()) {
257 LOG(ERROR) << "send num too high, wait:" << send_num_;
258 usleep(100);
259 continue;
260 }
261 if (batch_req.size() < config_.ClientBatchNum()) {
262 std::unique_ptr<QueueItem> item =
263 batch_queue_.Pop(config_.ClientBatchWaitTimeMS());
264 if (item != nullptr) {
265 batch_req.push_back(std::move(item));
266 if (batch_req.size() < config_.ClientBatchNum()) {
267 continue;
268 }
269 }
270 }
271 if (batch_req.empty()) {
272 continue;
273 }
274 int ret = DoBatch(batch_req);
275 batch_req.clear();
276 if (ret != 0) {
277 Response response;
278 response.set_result(Response::ERROR);
279 for (size_t i = 0; i < batch_req.size(); ++i) {
280 if (batch_req[i]->context && batch_req[i]->context->client) {
281 int ret = batch_req[i]->context->client->SendRawMessage(response);
282 if (ret) {
283 LOG(ERROR) << "send resp" << response.DebugString()
284 << " fail ret:" << ret;
285 }
286 }
287 }
288 }
289 }
290 return 0;
291}

293int ResponseManager::DoBatch(
294 const std::vector<std::unique_ptr<QueueItem>>& batch_req) {
295 auto new_request =
296 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
297 if (new_request == nullptr) {
298 return -2;
299 }
300 std::vector<std::unique_ptr<Context>> context_list;
301
302 BatchUserRequest batch_request;
303 for (size_t i = 0; i < batch_req.size(); ++i) {
304 BatchUserRequest::UserRequest* req = batch_request.add_user_requests();
305 *req->mutable_request() = *batch_req[i]->user_request.get();
306 *req->mutable_signature() = batch_req[i]->context->signature;
307 req->set_id(i);
308 context_list.push_back(std::move(batch_req[i]->context));
309 }
310
311 if (!config_.IsPerformanceRunning()) {
312 LOG(ERROR) << "add context list:" << new_request->seq()
313 << " list size:" << context_list.size()
314 << " local_id:" << local_id_;
315 batch_request.set_local_id(local_id_);
316 int ret = AddContextList(std::move(context_list), local_id_++);
317 if (ret != 0) {
318 LOG(ERROR) << "add context list fail:";
319 return ret;
320 }
321 }
322 batch_request.set_createtime(GetCurrentTime());
323 std::string data;
324 batch_request.SerializeToString(&data);
325 if (verifier_) {
326 auto signature_or = verifier_->SignMessage(data);
327 if (!signature_or.ok()) {
328 LOG(ERROR) << "Sign message fail";
329 return -2;
330 }
331 *new_request->mutable_data_signature() = *signature_or;
332 }
333
334 batch_request.SerializeToString(new_request->mutable_data());
335 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
336 new_request->set_proxy_id(config_.GetSelfInfo().id());
337 replica_communicator_->SendMessage(*new_request, GetPrimary());
338 send_num_++;
339 // LOG(INFO) << "send msg to primary:" << GetPrimary()
340 // << " batch size:" << batch_req.size();
341 AddWaitingResponseRequest(std::move(new_request));
342 return 0;
343}

345void ResponseManager::AddWaitingResponseRequest(
346 std::unique_ptr<Request> request) {
347 if (!config_.GetConfigData().enable_viewchange()) {
348 return;
349 }
350 pm_lock_.lock();
351 assert(timeout_length_ > 0);
352 uint64_t time = GetCurrentTime() + timeout_length_;
353 client_timeout_min_heap_.push(ResponseClientTimeout(request->hash(), time));
354 waiting_response_batches_.insert(
355 make_pair(request->hash(), std::move(request)));
356 pm_lock_.unlock();
357 sem_post(&request_sent_signal_);
358}

360void ResponseManager::RemoveWaitingResponseRequest(const std::string& hash) {
361 if (!config_.GetConfigData().enable_viewchange()) {
362 return;
363 }
364 pm_lock_.lock();
365 if (waiting_response_batches_.find(hash) != waiting_response_batches_.end()) {
366 waiting_response_batches_.erase(waiting_response_batches_.find(hash));
367 }
368 pm_lock_.unlock();
369}

386void ResponseManager::MonitoringClientTimeOut() {
387 while (!stop_) {
388 sem_wait(&request_sent_signal_);
389 pm_lock_.lock();
390 if (client_timeout_min_heap_.empty()) {
391 pm_lock_.unlock();
392 continue;
393 }
394 auto client_timeout = client_timeout_min_heap_.top();
395 client_timeout_min_heap_.pop();
396 pm_lock_.unlock();
397
398 if (client_timeout.timeout_time > GetCurrentTime()) {
399 usleep(client_timeout.timeout_time - GetCurrentTime());
400 }
401
402 if (CheckTimeOut(client_timeout.hash)) {
403 auto request = GetTimeOutRequest(client_timeout.hash);
404 if (request) {
405 replica_communicator_->BroadCast(*request);
406 }
407 }
408 }
