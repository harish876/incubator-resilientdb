47void Consensus::Init() {
48 if (performance_manager_ == nullptr) {
49 performance_manager_ =
50 config_.IsPerformanceRunning()
51 ? std::make_unique<PerformanceManager>(
52 config_, GetBroadCastClient(), GetSignatureVerifier())
53 : nullptr;
54 }
55
56 if (response_manager_ == nullptr) {
57 response_manager_ =
58 !config_.IsPerformanceRunning()
59 ? std::make_unique<ResponseManager>(config_, GetBroadCastClient(),
60 GetSignatureVerifier())
61 : nullptr;
62 }
63}

126int Consensus::ConsensusCommit(std::unique_ptr<Context> context,
127 std::unique_ptr<Request> request) {
128 switch (request->type()) {
129 case Request::TYPE_CLIENT_REQUEST:
130 if (config_.IsPerformanceRunning()) {
131 return performance_manager_->StartEval();
132 }
133 case Request::TYPE_RESPONSE:
134 if (config_.IsPerformanceRunning()) {
135 return performance_manager_->ProcessResponseMsg(std::move(context),
136 std::move(request));
137 }
138 case Request::TYPE_NEW_TXNS: {
139 return ProcessNewTransaction(std::move(request));
140 }
141 case Request::TYPE_CUSTOM_CONSENSUS: {
142 return ProcessCustomConsensus(std::move(request));
143 }
144 }
145 return 0;
146}

156int Consensus::ResponseMsg(const BatchUserResponse& batch_resp) {
157 Request request;
158 request.set_seq(batch_resp.seq());
159 request.set_type(Request::TYPE_RESPONSE);
160 request.set_sender_id(config_.GetSelfInfo().id());
161 request.set_proxy_id(batch_resp.proxy_id());
162 batch_resp.SerializeToString(request.mutable_data());
163 replica_communicator_->SendMessage(request, request.proxy_id());
164 return 0;
165}
