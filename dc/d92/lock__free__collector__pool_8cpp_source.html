27uint32_t GetCapacity(uint32_t size) {
28 int i = 0;
29 for (i = 0; (1u << i) <= size; ++i)
30 ;
31 return (1u << i);

35LockFreeCollectorPool::LockFreeCollectorPool(const std::string& name,
36 uint32_t size,
37 TransactionExecutor* executor,
38 bool enable_viewchange)
39 : name_(name),
40 capacity_(GetCapacity(size * 2)),
41 mask_((capacity_ << 1) - 1),
42 executor_(executor),
43 enable_viewchange_(enable_viewchange) {
44 collector_.resize(capacity_ << 1);
45 for (size_t i = 0; i < (capacity_ << 1); ++i) {
46 collector_[i] = std::make_unique<TransactionCollector>(i, executor_,
47 enable_viewchange_);
48 }
49 LOG(ERROR) << "name:" << name_ << " create pool done. capacity:" << capacity_
50 << " enable viewchange:" << enable_viewchange_ << " done";
51}

53void LockFreeCollectorPool::Update(uint64_t seq) {
54 uint32_t idx = seq & mask_;
55 if (collector_[idx]->Seq() != seq) {
56 LOG(ERROR) << "seq not match, skip update:" << seq;
57 return;
58 }
59 // LOG(ERROR)<<" update:"<<(idx^capacity_)<<" seq:"<<seq+capacity_<<
60 // " cap:"<<capacity_<<" update seq:"<<seq;
61 collector_[idx ^ capacity_] = std::make_unique<TransactionCollector>(
62 seq + capacity_, executor_, enable_viewchange_);
63}

65TransactionCollector* LockFreeCollectorPool::GetCollector(uint64_t seq) {
66 uint32_t idx = seq & mask_;
67 return collector_[idx].get();
68}
