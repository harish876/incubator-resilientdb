72int MemoryDB::SetValueWithVersion(const std::string& key,
73 const std::string& value, int version) {
74 auto it = kv_map_with_v_.find(key);
75 if ((it == kv_map_with_v_.end() && version != 0) ||
76 (it != kv_map_with_v_.end() && it->second.back().second != version)) {
77 LOG(ERROR) << " value version not match. key:" << key << " db version:"
78 << (it == kv_map_with_v_.end() ? 0 : it->second.back().second)
79 << " user version:" << version;
80 return -2;
81 }
82 kv_map_with_v_[key].push_back(std::make_pair(value, version + 1));
83 return 0;
84}

86std::pair<std::string, int> MemoryDB::GetValueWithVersion(
87 const std::string& key, int version) {
88 auto search_it = kv_map_with_v_.find(key);
89 if (search_it != kv_map_with_v_.end() && search_it->second.size()) {
90 auto it = search_it->second.end();
91 do {
92 --it;
93 if (it->second == version) {
94 return *it;
95 }
96 if (it->second < version) {
97 break;
98 }
99 } while (it != search_it->second.begin());
100 it = --search_it->second.end();
101 LOG(ERROR) << " key:" << key << " no version:" << version
102 << " return max:" << it->second;
103 return *it;
104 }
105 return std::make_pair("", 0);
106}

129std::vector<std::pair<std::string, int>> MemoryDB::GetHistory(
130 const std::string& key, int min_version, int max_version) {
131 std::vector<std::pair<std::string, int>> resp;
132 auto search_it = kv_map_with_v_.find(key);
133 if (search_it == kv_map_with_v_.end()) {
134 return resp;
135 }
136
137 auto it = search_it->second.end();
138 do {
139 --it;
140 if (it->second < min_version) {
141 break;
142 }
143 if (it->second <= max_version) {
144 resp.push_back(*it);
145 }
146 } while (it != search_it->second.begin());
147 return resp;
148}
