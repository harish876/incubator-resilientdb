50bool GeoPBFTCommitment::VerifyCerts(const BatchUserRequest& request,
51 const std::string& raw_data) {
52 if (verifier_) {
53 std::string hash = request.hash();
54 for (const auto& sig : request.committed_certs().committed_certs()) {
55 bool valid = verifier_->VerifyMessage(hash, sig);
56 if (!valid) {
57 LOG(ERROR) << "sign is not valid:" << sig.DebugString();
58 return false;
59 }
60 }
61 }
62 return true;
63}

82int GeoPBFTCommitment::GeoProcessCcm(std::unique_ptr<Context> context,
83 std::unique_ptr<Request> request) {
84 int sender_region_id = request->region_info().region_id();
85 if (!AddNewReq(request->seq(), sender_region_id)) {
86 // LOG(ERROR) << "geo_request already received, from sender id: "
87 // << request->sender_id() << " seq:" << request->seq() <<"
88 // region:"<<sender_region_id
89 // << " ,no more action needed.";
90 return 1;
91 }
92 // global_stats_->IncGeoRequest();
93
94 BatchUserRequest batch_request;
95 if (!batch_request.ParseFromString(request->data())) {
96 LOG(ERROR) << "[GeoGlobalExecutor] parse data fail!";
97 return -2;
98 }
99
100 if (!batch_request.has_committed_certs() ||
101 !VerifyCerts(batch_request, request->data())) {
102 // CheckCertificates
103 LOG(ERROR) << "no certs";
104 return -2;
105 }
106 // return global_executor_->OrderGeoRequest(std::move(request));
107
108 ResConfigData config_data = config_.GetConfigData();
109 int self_region_id = config_data.self_region_id();
110 // LOG(ERROR)<<"get request seq:"<<request->seq()<<" from:"<<sender_region_id;
111 // if the request comes from another region, do local broadcast
112 if (sender_region_id != self_region_id) {
113 std::unique_ptr<Request> broadcast_geo_req =
114 resdb::NewRequest(Request::TYPE_GEO_REQUEST, *request,
115 config_.GetSelfInfo().id(), sender_region_id);
116 // LOG(ERROR) << "[GeoProcessCcm] start broadcasting geo_request locally...
117 // ";
118 replica_communicator_->BroadCast(*broadcast_geo_req);
119 }
120 return global_executor_->OrderGeoRequest(std::move(request));
121}

123int GeoPBFTCommitment::PostProcessExecutedMsg() {
124 while (!stop_) {
125 auto batch_resp = global_executor_->GetResponseMsg();
126 if (batch_resp == nullptr) {
127 continue;
128 }
129 Request request;
130 request.set_seq(batch_resp->seq());
131 request.set_type(Request::TYPE_RESPONSE);
132 request.set_sender_id(config_.GetSelfInfo().id());
133 request.set_current_view(batch_resp->current_view());
134 request.set_proxy_id(batch_resp->proxy_id());
135 UpdateSeq(batch_resp->seq());
136 batch_resp->SerializeToString(request.mutable_data());
137 replica_communicator_->SendMessage(request, request.proxy_id());
138 }
139 return 0;
140}
