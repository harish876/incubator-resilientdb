99int64_t GetRSS() {
100 int64_t rss = 0;
101 FILE* fp = NULL;
102 if ((fp = fopen("/proc/self/statm", "r")) == NULL) {
103 return 0;
104 }
105
106 uint64_t size, resident, share, text, lib, data, dt;
107 if (fscanf(fp, "%lu %lu %lu %lu %lu %lu %lu", &size, &resident, &share, &text,
108 &lib, &data, &dt) != 7) {
109 fclose(fp);
110 return 0;
111 }
112 fclose(fp);
113
114 int64_t page_size = sysconf(_SC_PAGESIZE);
115 rss = resident * page_size;
116
117 // Convert to MB
118 rss = rss / (1024 * 1024);
119
120 return rss;
121}

216 }

266void Stats::GetTransactionDetails(BatchUserRequest batch_request) {
267 if (!enable_resview) {
268 return;
269 }
270 transaction_summary_.txn_number = batch_request.seq();
271 transaction_summary_.txn_command.clear();
272 transaction_summary_.txn_key.clear();
273 transaction_summary_.txn_value.clear();
274 for (auto& sub_request : batch_request.user_requests()) {
275 KVRequest kv_request;
276 if (!kv_request.ParseFromString(sub_request.request().data())) {
277 break;
278 }
279 if (kv_request.cmd() == KVRequest::SET) {
280 transaction_summary_.txn_command.push_back("SET");
281 transaction_summary_.txn_key.push_back(kv_request.key());
282 transaction_summary_.txn_value.push_back(kv_request.value());
283 } else if (kv_request.cmd() == KVRequest::GET) {
284 transaction_summary_.txn_command.push_back("GET");
285 transaction_summary_.txn_key.push_back(kv_request.key());
286 transaction_summary_.txn_value.push_back("");
287 } else if (kv_request.cmd() == KVRequest::GETALLVALUES) {
288 transaction_summary_.txn_command.push_back("GETALLVALUES");
289 transaction_summary_.txn_key.push_back(kv_request.key());
290 transaction_summary_.txn_value.push_back("");
291 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
292 transaction_summary_.txn_command.push_back("GETRANGE");
293 transaction_summary_.txn_key.push_back(kv_request.key());
294 transaction_summary_.txn_value.push_back(kv_request.value());
295 }
296 }

299void Stats::SendSummary() {
300 if (!enable_resview) {
301 return;
302 }
303 transaction_summary_.execution_time = std::chrono::system_clock::now();
304
305 // Convert Transaction Summary to JSON
306 summary_json_["replica_id"] = transaction_summary_.replica_id;
307 summary_json_["ip"] = transaction_summary_.ip;
308 summary_json_["port"] = transaction_summary_.port;
309 summary_json_["primary_id"] = transaction_summary_.primary_id;
310 summary_json_["propose_pre_prepare_time"] =
311 transaction_summary_.request_pre_prepare_state_time.time_since_epoch()
312 .count();
313 summary_json_["prepare_time"] =
314 transaction_summary_.prepare_state_time.time_since_epoch().count();
315 summary_json_["commit_time"] =
316 transaction_summary_.commit_state_time.time_since_epoch().count();
317 summary_json_["execution_time"] =
318 transaction_summary_.execution_time.time_since_epoch().count();
319 for (size_t i = 0;
320 i < transaction_summary_.prepare_message_count_times_list.size(); i++) {
321 summary_json_["prepare_message_timestamps"].push_back(
322 transaction_summary_.prepare_message_count_times_list[i]
323 .time_since_epoch()
324 .count());
325 }
326 for (size_t i = 0;
327 i < transaction_summary_.commit_message_count_times_list.size(); i++) {
328 summary_json_["commit_message_timestamps"].push_back(
329 transaction_summary_.commit_message_count_times_list[i]
330 .time_since_epoch()
331 .count());
332 }
333 summary_json_["txn_number"] = transaction_summary_.txn_number;
334 for (size_t i = 0; i < transaction_summary_.txn_command.size(); i++) {
335 summary_json_["txn_commands"].push_back(
336 transaction_summary_.txn_command[i]);
337 }
338 for (size_t i = 0; i < transaction_summary_.txn_key.size(); i++) {
339 summary_json_["txn_keys"].push_back(transaction_summary_.txn_key[i]);
340 }
341 for (size_t i = 0; i < transaction_summary_.txn_value.size(); i++) {
342 summary_json_["txn_values"].push_back(transaction_summary_.txn_value[i]);
343 }
344
345 summary_json_["ext_cache_hit_ratio"] =
346 transaction_summary_.ext_cache_hit_ratio_;
347 consensus_history_[std::to_string(transaction_summary_.txn_number)] =
348 summary_json_;
349
350 LOG(ERROR) << summary_json_.dump();
351
352 // Reset Transaction Summary Parameters
353 transaction_summary_.request_pre_prepare_state_time =
354 std::chrono::system_clock::time_point::min();
355 transaction_summary_.prepare_state_time =
356 std::chrono::system_clock::time_point::min();
357 transaction_summary_.commit_state_time =
358 std::chrono::system_clock::time_point::min();
359 transaction_summary_.execution_time =
360 std::chrono::system_clock::time_point::min();
361 transaction_summary_.prepare_message_count_times_list.clear();
362 transaction_summary_.commit_message_count_times_list.clear();
363
364 summary_json_.clear();
365}
