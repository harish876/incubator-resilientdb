74// =================== response ========================
75// handle the response message. If receive f+1 commit messages, send back to the
76// caller.
77int ResponseManager::ProcessResponseMsg(std::unique_ptr<Context> context,
78 std::unique_ptr<Request> request) {
79 std::unique_ptr<Request> response;
80 // Add the response message, and use the call back to collect the received
81 // messages.
82 // The callback will be triggered if it received f+1 messages.
83 if (request->ret() == -2) {
84 LOG(ERROR) << "get response fail:" << request->ret();
85 send_num_--;
86 return 0;
87 }
88 CollectorResultCode ret =
89 AddResponseMsg(std::move(request), [&](const Request& request) {
90 response = std::make_unique<Request>(request);
91 return;
92 });
93
94 if (ret == CollectorResultCode::STATE_CHANGED) {
95 BatchUserResponse batch_response;
96 if (batch_response.ParseFromString(response->data())) {
97 SendResponseToClient(batch_response);
98 } else {
99 LOG(ERROR) << "parse response fail:";
100 }
101 }
102 return ret == CollectorResultCode::INVALID ? -2 : 0;
103}

105CollectorResultCode ResponseManager::AddResponseMsg(
106 std::unique_ptr<Request> request,
107 std::function<void(const Request&)> response_call_back) {
108 if (request == nullptr) {
109 return CollectorResultCode::INVALID;
110 }
111
112 int type = request->type();
113 uint64_t seq = request->seq();
114 bool done = false;
115 {
116 int idx = seq % response_set_size_;
117 std::unique_lock<std::mutex> lk(response_lock_[idx]);
118 if (response_[idx][seq] == -1) {
119 return CollectorResultCode::OK;
120 }
121 response_[idx][seq]++;
122 if (response_[idx][seq] >= config_.GetMinClientReceiveNum()) {
123 response_[idx][seq] = -1;
124 done = true;
125 }
126 }
127 if (done) {
128 response_call_back(*request);
129 return CollectorResultCode::STATE_CHANGED;
130 }
131 return CollectorResultCode::OK;
132}

134void ResponseManager::SendResponseToClient(
135 const BatchUserResponse& batch_response) {
136 uint64_t create_time = batch_response.createtime();
137 uint64_t local_id = batch_response.local_id();
138 if (create_time > 0) {
139 uint64_t run_time = GetCurrentTime() - create_time;
140 global_stats_->AddLatency(run_time);
141 } else {
142 LOG(ERROR) << "seq:" << local_id << " no resp";
143 }
144 send_num_--;
145}

147// =================== request ========================
148int ResponseManager::BatchProposeMsg() {
149 LOG(INFO) << "batch wait time:" << config_.ClientBatchWaitTimeMS()
150 << " batch num:" << config_.ClientBatchNum();
151 std::vector<std::unique_ptr<QueueItem>> batch_req;
152 while (!stop_) {
153 if (send_num_ > config_.GetMaxProcessTxn()) {
154 LOG(ERROR) << "send num too high, wait:" << send_num_;
155 usleep(100);
156 continue;
157 }
158 if (batch_req.size() < config_.ClientBatchNum()) {
159 std::unique_ptr<QueueItem> item =
160 batch_queue_.Pop(config_.ClientBatchWaitTimeMS());
161 if (item != nullptr) {
162 batch_req.push_back(std::move(item));
163 if (batch_req.size() < config_.ClientBatchNum()) {
164 continue;
165 }
166 }
167 }
168 if (batch_req.empty()) {
169 continue;
170 }
171 int ret = DoBatch(batch_req);
172 batch_req.clear();
173 if (ret != 0) {
174 Response response;
175 response.set_result(Response::ERROR);
176 for (size_t i = 0; i < batch_req.size(); ++i) {
177 if (batch_req[i]->context && batch_req[i]->context->client) {
178 int ret = batch_req[i]->context->client->SendRawMessage(response);
179 if (ret) {
180 LOG(ERROR) << "send resp" << response.DebugString()
181 << " fail ret:" << ret;
182 }
183 }
184 }
185 }
186 }
187 return 0;
188}

190int ResponseManager::DoBatch(
191 const std::vector<std::unique_ptr<QueueItem>>& batch_req) {
192 auto new_request =
193 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
194 if (new_request == nullptr) {
195 return -2;
196 }
197 std::vector<std::unique_ptr<Context>> context_list;
198
199 BatchUserRequest batch_request;
200 for (size_t i = 0; i < batch_req.size(); ++i) {
201 BatchUserRequest::UserRequest* req = batch_request.add_user_requests();
202 *req->mutable_request() = *batch_req[i]->user_request.get();
203 *req->mutable_signature() = batch_req[i]->context->signature;
204 req->set_id(i);
205 context_list.push_back(std::move(batch_req[i]->context));
206 }
207
208 if (!config_.IsPerformanceRunning()) {
209 LOG(ERROR) << "add context list:" << new_request->seq()
210 << " list size:" << context_list.size();
211 batch_request.set_local_id(local_id_);
212 }
213 batch_request.set_createtime(GetCurrentTime());
214 std::string data;
215 batch_request.SerializeToString(&data);
216 if (verifier_) {
217 auto signature_or = verifier_->SignMessage(data);
218 if (!signature_or.ok()) {
219 LOG(ERROR) << "Sign message fail";
220 return -2;
221 }
222 *new_request->mutable_data_signature() = *signature_or;
223 }
224
225 batch_request.SerializeToString(new_request->mutable_data());
226 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
227 new_request->set_proxy_id(config_.GetSelfInfo().id());
228 replica_communicator_->SendMessage(*new_request, GetPrimary());
229 send_num_++;
230 LOG(INFO) << "send msg to primary:" << GetPrimary()
231 << " batch size:" << batch_req.size();
232 return 0;
233}
