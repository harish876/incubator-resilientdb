34def load_tool_map() -> list[dict]:
35 tool_map_path = REPO_ROOT / "ecosystem" / "pocketflow" / "tool-doc-map.json"
36 try:
37 with tool_map_path.open("r", encoding="utf-8") as fh:

43def find_tool_entry(tool_name: str, tool_map: list[dict]) -> dict:

50def ensure_doc_file(path: Path, tool_name: str) -> None:
51 if path.exists():
52 return

54 path.parent.mkdir(parents=True, exist_ok=True)
55 template = DEFAULT_TEMPLATE.format(name=tool_name, title=tool_name.title())
56 with path.open("w", encoding="utf-8") as fh:
57 fh.write(template.strip() + "\n")

61def replace_block(original: str, tool_name: str, body: str) -> str:
62 start_marker = f"{{/* BEGIN AUTO_DOC: {tool_name} */}}"
63 end_marker = f"{{/* END AUTO_DOC: {tool_name} */}}"
64
65 start_idx = original.find(start_marker)
66 end_idx = original.find(end_marker)
67
68 if start_idx == -1 or end_idx == -1:
69 raise SystemExit(
70 f"Markers not found in MDX file. Expected '{start_marker}' and '{end_marker}'."
71 )
72 if end_idx <= start_idx:
73 raise SystemExit("END marker appears before BEGIN marker.")
74
75 before = original[: start_idx + len(start_marker)]
76 after = original[end_idx:]
77
78 body_normalized = "\n\n" + body.strip() + "\n\n"
79 return before + body_normalized + after

82def run_prettier(path: Path) -> None:
83 cmd = ["npx", "prettier", "--write", str(path)]

98def main() -> None:
99 parser = argparse.ArgumentParser(description="Update MDX auto-doc section.")
100 parser.add_argument("--tool", required=True, help="Tool name in tool-doc-map.json")
101 parser.add_argument(
102 "--body-file", required=True, help="Path to file containing markdown body"
103 )
104 parser.add_argument(
105 "--skip-format",
106 action="store_true",
107 help="Skip running Prettier after updating the MDX file.",
108 )
109 args = parser.parse_args()
110
111 body_path = Path(args.body_file)
112 if not body_path.exists():
113 raise SystemExit(f"Body file not found: {body_path}")
114
115 tool_map = load_tool_map()
116 entry = find_tool_entry(args.tool, tool_map)
117
118 doc_rel_path = entry.get("doc_path")
119 if not doc_rel_path:
120 raise SystemExit(f"'doc_path' missing for tool {args.tool}")
121 doc_path = REPO_ROOT / doc_rel_path
122
123 ensure_doc_file(doc_path, args.tool)
124 body = body_path.read_text(encoding="utf-8")
125 original = doc_path.read_text(encoding="utf-8")
126 updated = replace_block(original, args.tool, body)
127 doc_path.write_text(updated, encoding="utf-8")
128 print(f"[update-mdx] Updated {doc_path}")
129
130 if not args.skip_format:
131 run_prettier(doc_path)
132
