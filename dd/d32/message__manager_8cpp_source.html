28MessageManager::MessageManager(
29 const ResDBConfig& config,
30 std::unique_ptr<TransactionManager> transaction_manager,
31 CheckPointManager* checkpoint_manager, SystemInfo* system_info)
32 : config_(config),
33 queue_("executed"),
34 txn_db_(checkpoint_manager->GetTxnDB()),
35 system_info_(system_info),
36 checkpoint_manager_(checkpoint_manager),
37 transaction_executor_(std::make_unique<TransactionExecutor>(
38 config,
39 [&](std::unique_ptr<Request> request,
40 std::unique_ptr<BatchUserResponse> resp_msg) {
41 if (request->is_recovery()) {
42 if (checkpoint_manager_) {
43 checkpoint_manager_->AddCommitData(std::move(request));
44 }
45 return;
46 }
47 resp_msg->set_proxy_id(request->proxy_id());
48 resp_msg->set_seq(request->seq());
49 resp_msg->set_current_view(request->current_view());
50 resp_msg->set_primary_id(GetCurrentPrimary());
51 if (transaction_executor_->NeedResponse() &&
52 resp_msg->proxy_id() != 0) {
53 queue_.Push(std::move(resp_msg));
54 }
55 if (checkpoint_manager_) {
56 checkpoint_manager_->AddCommitData(std::move(request));
57 }
58 },
59 system_info_, std::move(transaction_manager))),
60 collector_pool_(std::make_unique<LockFreeCollectorPool>(
61 "txn", config_.GetMaxProcessTxn(), transaction_executor_.get(),
62 config_.GetConfigData().enable_viewchange())) {
63 global_stats_ = Stats::GetGlobalStats();
64 transaction_executor_->SetSeqUpdateNotifyFunc(
65 [&](uint64_t seq) { collector_pool_->Update(seq - 1); });
66 checkpoint_manager_->SetExecutor(transaction_executor_.get());
67}

94absl::StatusOr<uint64_t> MessageManager::AssignNextSeq() {
95 std::unique_lock<std::mutex> lk(seq_mutex_);
96 uint32_t max_executed_seq = transaction_executor_->GetMaxPendingExecutedSeq();
97 global_stats_->SeqGap(next_seq_ - max_executed_seq);
98 if (next_seq_ - max_executed_seq >
99 static_cast<uint64_t>(config_.GetMaxProcessTxn())) {
100 // LOG(ERROR) << "next_seq_: " << next_seq_ << " max_executed_seq: " <<
101 // max_executed_seq;
102 return absl::InvalidArgumentError("Seq has been used up.");
103 }
104 return next_seq_++;
105}

111// Check if the request is valid.
112// 1. view is the same as the current view
113// 2. seq is larger or equal than the next execute seq.
114// 3. inside the water mark.
115bool MessageManager::IsValidMsg(const Request& request) {
116 if (request.type() == Request::TYPE_RESPONSE) {
117 return true;
118 }
119 // view should be the same as the current one.
120 if (static_cast<uint64_t>(request.current_view()) != GetCurrentView()) {
121 LOG(ERROR) << "message view :[" << request.current_view()
122 << "] is older than the cur view :[" << GetCurrentView() << "]";
123 return false;
124 }
125
126 if (static_cast<uint64_t>(request.seq()) <
127 transaction_executor_->GetMaxPendingExecutedSeq()) {
128 return false;
129 }
130
131 return true;
132}

134bool MessageManager::MayConsensusChangeStatus(
135 int type, int received_count, std::atomic<TransactionStatue>* status,
136 bool ret) {
137 switch (type) {
138 case Request::TYPE_PRE_PREPARE:
139 if (*status == TransactionStatue::None) {
140 TransactionStatue old_status = TransactionStatue::None;
141 return status->compare_exchange_strong(
142 old_status, TransactionStatue::READY_PREPARE,
143 std::memory_order_acq_rel, std::memory_order_acq_rel);
144 }
145 break;
146 case Request::TYPE_PREPARE:
147 if (*status == TransactionStatue::READY_PREPARE &&
148 config_.GetMinDataReceiveNum() <= received_count) {
149 TransactionStatue old_status = TransactionStatue::READY_PREPARE;
150 return status->compare_exchange_strong(
151 old_status, TransactionStatue::READY_COMMIT,
152 std::memory_order_acq_rel, std::memory_order_acq_rel);
153 }
154 break;
155 case Request::TYPE_COMMIT:
156 if (*status == TransactionStatue::READY_COMMIT &&
157 config_.GetMinDataReceiveNum() <= received_count) {
158 TransactionStatue old_status = TransactionStatue::READY_COMMIT;
159 return status->compare_exchange_strong(
160 old_status, TransactionStatue::READY_EXECUTE,
161 std::memory_order_acq_rel, std::memory_order_acq_rel);
162 return true;
163 }
164 break;
165 }
166 return ret;
167}

169// Add commit messages and return the number of messages have been received.
170// The commit messages only include post(pre-prepare), prepare and commit
171// messages. Messages are handled by state (PREPARE,COMMIT,READY_EXECUTE).
172
173// If there are enough messages and the state is changed after adding the
174// message, return 1, otherwise return 0. Return -2 if the request is not valid.
175CollectorResultCode MessageManager::AddConsensusMsg(
176 const SignatureInfo& signature, std::unique_ptr<Request> request) {
177 if (request == nullptr || !IsValidMsg(*request)) {
178 return CollectorResultCode::INVALID;
179 }
180 int type = request->type();
181 uint64_t seq = request->seq();
182 int resp_received_count = 0;
183 int proxy_id = request->proxy_id();
184
185 int ret = collector_pool_->GetCollector(seq)->AddRequest(
186 std::move(request), signature, type == Request::TYPE_PRE_PREPARE,
187 [&](const Request& request, int received_count,
188 TransactionCollector::CollectorDataType* data,
189 std::atomic<TransactionStatue>* status, bool force) {
190 if (MayConsensusChangeStatus(type, received_count, status, force)) {
191 resp_received_count = 1;
192 }
193 });
194 if (ret == 1) {
195 SetLastCommittedTime(proxy_id);
196 } else if (ret != 0) {
197 return CollectorResultCode::INVALID;
198 }
199 if (resp_received_count > 0) {
200 return CollectorResultCode::STATE_CHANGED;
201 }
202 return CollectorResultCode::OK;
203}

205RequestSet MessageManager::GetRequestSet(uint64_t min_seq, uint64_t max_seq) {
206 RequestSet ret;
207 std::unique_lock<std::mutex> lk(data_mutex_);
208 for (uint64_t i = min_seq; i <= max_seq; ++i) {
209 if (committed_data_.find(i) == committed_data_.end()) {
210 LOG(ERROR) << "seq :" << i << " doesn't exist";
211 continue;
212 }
213 RequestWithProof* request = ret.add_requests();
214 *request->mutable_request() = committed_data_[i];
215 request->set_seq(i);
216 for (const auto& request_info : committed_proof_[i]) {
217 RequestWithProof::RequestData* data = request->add_proofs();
218 *data->mutable_request() = *request_info->request;
219 *data->mutable_signature() = request_info->signature;
220 }
221 }
222 return ret;
223}

274void MessageManager::SendResponse(std::unique_ptr<Request> request) {
275 std::unique_ptr<BatchUserResponse> response =
276 std::make_unique<BatchUserResponse>();
277 response->set_createtime(GetCurrentTime());
278 // response->set_local_id(batch_request.local_id());
279 response->set_hash(request->hash());
280 response->set_proxy_id(request->proxy_id());
281 response->set_seq(request->seq());
282 response->set_current_view(GetCurrentView());
283 response->set_primary_id(GetCurrentPrimary());
284 if (transaction_executor_->NeedResponse() && response->proxy_id() != 0) {
285 queue_.Push(std::move(response));
286 }
