93absl::Status PoWManager::WaitBlockDone() {
94 LOG(ERROR) << "wait block:" << current_status_;
95 int timeout_ms = config_.GetMiningTime();
96 // 60000000;
97 // int timeout_ms = 300000000;
98 std::unique_lock<std::mutex> lk(mutex_);
99 cv_.wait_for(lk, std::chrono::microseconds(timeout_ms),
100 [&] { return current_status_ == NEXT_NEWBLOCK; });
101 LOG(ERROR) << "wait block done:" << current_status_;
102 if (current_status_ == NEXT_NEWBLOCK) {
103 return absl::OkStatus();
104 }
105 return absl::NotFoundError("No new transaction.");
106}

141int PoWManager::GetMiningTxn(MiningType type) {
142 std::unique_lock<std::mutex> lck(tx_mutex_);
143 LOG(ERROR) << "get mining txn status:" << current_status_;
144 if (current_status_ == NEXT_NEWBLOCK) {
145 type = MiningType::NEWBLOCK;
146 }
147 if (type == NEWBLOCK) {
148 uint64_t max_seq = std::max(block_manager_->GetLastSeq(),
149 block_manager_->GetLastCandidateSeq());
150 LOG(ERROR) << "get block last max:" << block_manager_->GetLastSeq() << " "
151 << block_manager_->GetLastCandidateSeq();
152 auto client_tx = transaction_accessor_->ConsumeTransactions(max_seq + 1);
153 if (client_tx == nullptr) {
154 return -2;
155 }
156 block_manager_->SetNewMiningBlock(std::move(client_tx));
157 } else {
158 // prometheus_handler_->Inc(SHIFT_MSG, 1);
159 int ret = GetShiftMsg(need_slice_info_);
160 LOG(ERROR) << "get shift msg ret:" << ret;
161 if (ret == 1) {
162 // no solution after enought shift.
163 return 1;
164 }
165 if (ret != 0) {
166 BroadCastShiftMsg(need_slice_info_);
167 return -2;
168 }
169 LOG(ERROR) << "get shift msg:" << need_slice_info_.DebugString();
170 }
171 return 0;
172}

174PoWManager::MiningStatus PoWManager::Wait() {
175 current_status_ = GENERATE_NEW;
176 auto mining_thread = std::thread([&]() {
177 LOG(ERROR) << "mine";
178 absl::Status status = block_manager_->Mine();
179 if (status.ok()) {
180 if (block_manager_->Commit() == 0) {
181 NotifyNextBlock();
182 }
183 LOG(ERROR) << "done mine";
184 }
185 LOG(ERROR) << "mine:" << status.ok();
186 });
187
188 auto status = WaitBlockDone();
189 if (mining_thread.joinable()) {
190 mining_thread.join();
191 }
192 LOG(ERROR) << "success:" << status.ok() << " status:" << current_status_;
193 if (status.ok() || current_status_ == NEXT_NEWBLOCK) {
194 return MiningStatus::OK;
195 }
196 return MiningStatus::TIMEOUT;
197}

226// Broadcast the new block if once it is committed.
227void PoWManager::GossipProcess() {
228 uint64_t last_height = 0;
229 while (IsRunning()) {
230 std::unique_lock<std::mutex> lck(broad_cast_mtx_);
231 broad_cast_cv_.wait_until(
232 lck, std::chrono::system_clock::now() + std::chrono::seconds(1));
233
234 Block* block = block_manager_->GetBlockByHeight(last_height + 1);
235 if (block == nullptr) {
236 // LOG(ERROR) << "======= get block fail:" << last_height
237 // << " id:" << config_.GetSelfInfo().id();
238 continue;
239 }
240 BroadCastNewBlock(*block);
241 last_height++;
242 }
243}

250// Mining the new blocks got from PBFT cluster.
251void PoWManager::MiningProcess() {
252 if (config_.GetSelfInfo().id() > 8 && config_.GetSelfInfo().id() < 13) {
253 // return;
254 }
255 LOG(ERROR) << "start";
256 MiningType type = MiningType::NEWBLOCK;
257 while (IsRunning()) {
258 int ret = GetMiningTxn(type);
259 LOG(ERROR) << "get mining ret:" << ret;
260 if (ret < 0) {
261 usleep(10000);
262 continue;
263 } else if (ret > 0) {
264 LOG(ERROR) << "get new block";
265 type = MiningType::NEWBLOCK;
266 continue;
267 }
268 LOG(ERROR) << "get ok";
269
270 auto mining_status = Wait();
271 LOG(ERROR) << "done:" << mining_status;
272 if (mining_status == MiningStatus::TIMEOUT) {
273 type = MiningType::SHIFTING;
274 SendShiftMsg();
275 } else {
276 type = MiningType::NEWBLOCK;
277 LOG(ERROR) << "done";
278 NotifyBroadCast();
279 }
280 }
