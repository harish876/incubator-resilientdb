2"""
3Run Pocketflow for a single tool and update its Beacon MDX page
4using only the generated index.md content.
5
6Usage (from repo root):
7
8 python ecosystem/pocketflow/scripts/run_pocketflow_for_tool.py --tool rescontract
9
10This will:
11 - Look up `rescontract` in tool-doc-map.json
12 - Run Pocketflow against its `code_root`
13 - Read the generated `index.md` for that project
14 - Inject the content into the MDX auto-doc block for that tool

33def load_tool_map() -> list[dict]:
34 tool_map_path = POCKETFLOW_DIR / "tool-doc-map.json"
35 try:
36 with tool_map_path.open("r", encoding="utf-8") as fh:
37 return json.load(fh)
38 except FileNotFoundError as exc:
39 raise SystemExit(f"tool map not found at {tool_map_path}") from exc

42def find_tool_entry(tool_name: str, tool_map: list[dict]) -> dict:
43 for entry in tool_map:
44 if entry.get("name") == tool_name:
45 return entry
46 raise SystemExit(f"tool '{tool_name}' missing from tool-doc-map.json")

49def run_pocketflow_for_dir(
50 code_root: Path,
51 output_dir: Path,
52 language: str = "english",
53 use_cache: bool = True,
54 max_abstractions: int = 10,
55 single_file: bool = False,
56) -> Path:
57 """
58 Invoke ecosystem/pocketflow/main.py pointing at code_root and
59 returning the expected path to index.md for the generated project.
60 """
61 main_py = POCKETFLOW_DIR / "main.py"
62 if not main_py.exists():
63 raise SystemExit(f"Pocketflow main.py not found at {main_py}")
64
65 project_name = os.path.basename(str(code_root))
66 output_dir.mkdir(parents=True, exist_ok=True)
67
68 cmd = [
69 sys.executable,
70 str(main_py),
71 "--dir",
72 str(code_root),
73 "-o",
74 str(output_dir),
75 "--language",
76 language,
77 "--max-abstractions",
78 str(max_abstractions),
79 ]
80 if single_file:
81 cmd.append("--single-file") # Generate single consolidated file instead of index + chapters
82 if not use_cache:
83 cmd.append("--no-cache")
84
85 print(f"[run-pf] Running Pocketflow for {code_root} -> {output_dir}")
86 try:
87 completed = subprocess.run(
88 cmd, cwd=REPO_ROOT, text=True, check=True
89 )
90 if completed.stdout:
91 print(completed.stdout)
92 if completed.stderr:
93 print(completed.stderr)
94 except subprocess.CalledProcessError as exc:
95 raise SystemExit(f"Pocketflow failed with exit code {exc.returncode}") from exc
96
97 index_path = output_dir / project_name / "index.md"
98 if not index_path.exists():
99 raise SystemExit(f"Pocketflow output index.md not found at {index_path}")
100 return index_path
