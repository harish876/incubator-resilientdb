30ReplicaCommunicator::ReplicaCommunicator(
31 const std::vector<ReplicaInfo>& replicas, SignatureVerifier* verifier,
32 bool is_use_long_conn, int epoll_num, int tcp_batch)
33 : replicas_(replicas),
34 verifier_(verifier),
35 is_running_(false),
36 batch_queue_("bc_batch", tcp_batch),
37 is_use_long_conn_(is_use_long_conn),
38 tcp_batch_(tcp_batch) {
39 global_stats_ = Stats::GetGlobalStats();
40 if (is_use_long_conn_) {
41 worker_ = std::make_unique<boost::asio::io_service::work>(io_service_);
42 for (int i = 0; i < epoll_num; ++i) {
43 worker_threads_.push_back(std::thread([&]() { io_service_.run(); }));
44 }
45 }
46 LOG(ERROR)<<" tcp batch:"<<tcp_batch;
47
48 StartBroadcastInBackGround();
49
50}

52ReplicaCommunicator::~ReplicaCommunicator() {
53 is_running_ = false;
54 if (broadcast_thread_.joinable()) {
55 broadcast_thread_.join();
56 }
57 if (is_use_long_conn_) {
58 for (auto& cli : client_pools_) {
59 cli.second.reset();
60 }
61 worker_.reset();
62 worker_ = nullptr;
63 io_service_.stop();
64 for (auto& worker_th : worker_threads_) {
65 if (worker_th.joinable()) {
66 worker_th.join();
67 }
68 }
69 }

104void ReplicaCommunicator::StartBroadcastInBackGround() {
105 is_running_ = true;
106 broadcast_thread_ = std::thread([&]() {
107 while (IsRunning()) {
108 std::vector<std::unique_ptr<QueueItem>> batch_req =
109 batch_queue_.Pop(10000);
110 if (batch_req.empty()) {
111 continue;
112 }
113 BroadcastData broadcast_data;
114 for (auto& queue_item : batch_req) {
115 broadcast_data.add_data()->swap(queue_item->data);
116 }
117
118 global_stats_->SendBroadCastMsg(broadcast_data.data_size());
119 int ret = SendMessageFromPool(broadcast_data, replicas_);
120 if (ret < 0) {
121 LOG(ERROR) << "broadcast request fail:";
122 }
123 }
124 });
125}

127void ReplicaCommunicator::StartSingleInBackGround(const std::string& ip, int port) {
128 single_bq_[std::make_pair(ip,port)] = std::make_unique<BatchQueue<std::unique_ptr<QueueItem>>>("s_batch", tcp_batch_);
129
130 ReplicaInfo replica_info;
131 for (const auto& replica : replicas_) {
132 if (replica.ip() == ip && replica.port() == port) {
133 replica_info = replica;
134 break;
135 }
136 }
137
138 if (replica_info.ip().empty()) {
139 for (const auto& replica : GetClientReplicas()) {
140 if (replica.ip() == ip && replica.port() == port) {
141 replica_info = replica;
142 break;
143 }
144 }
145 }
146
147
148 single_thread_.push_back(std::thread([&](BatchQueue<std::unique_ptr<QueueItem>> *bq, ReplicaInfo replica_info) {
149 while (IsRunning()) {
150 std::vector<std::unique_ptr<QueueItem>> batch_req =
151 bq->Pop(50000);
152 if (batch_req.empty()) {
153 continue;
154 }
155 BroadcastData broadcast_data;
156 for (auto& queue_item : batch_req) {
157 broadcast_data.add_data()->swap(queue_item->data);
158 }
159
160 global_stats_->SendBroadCastMsg(broadcast_data.data_size());
161 //LOG(ERROR)<<" send to ip:"<<replica_info.ip()<<" port:"<<replica_info.port()<<" bq size:"<<batch_req.size();
162 int ret = SendMessageFromPool(broadcast_data, {replica_info});
163 if (ret < 0) {
164 LOG(ERROR) << "broadcast request fail:";
165 }
166 //LOG(ERROR)<<" send to ip:"<<replica_info.ip()<<" port:"<<replica_info.port()<<" bq size:"<<batch_req.size()<<" done";
167 }
168 }, single_bq_[std::make_pair(ip,port)].get(), replica_info));
169}

172int ReplicaCommunicator::SendSingleMessage(const google::protobuf::Message& message,
173const ReplicaInfo& replica_info) {
174
175 std::string ip = replica_info.ip();
176 int port = replica_info.port();
177
178 //LOG(ERROR)<<" send msg ip:"<<ip<<" port:"<<port;
179 global_stats_->BroadCastMsg();
180 if (is_use_long_conn_) {
181 auto item = std::make_unique<QueueItem>();
182 item->data = NetChannel::GetRawMessageString(message, verifier_);
183 std::lock_guard<std::mutex> lk(smutex_);
184 if(single_bq_.find(std::make_pair(ip, port)) == single_bq_.end()){
185 StartSingleInBackGround(ip, port);
186 }
187 assert(single_bq_[std::make_pair(ip, port)] != nullptr);
188 single_bq_[std::make_pair(ip, port)]->Push(std::move(item));
189 return 0;
190 } else {
191 return SendMessageInternal(message, replicas_);
192 }
193}

221int ReplicaCommunicator::SendBatchMessage(
222 const std::vector<std::unique_ptr<Request>>& messages,
223 const ReplicaInfo& replica_info) {
224 if (is_use_long_conn_) {
225 BroadcastData broadcast_data;
226 for (const auto& message : messages) {
227 std::string data = NetChannel::GetRawMessageString(*message, verifier_);
228 broadcast_data.add_data()->swap(data);
229 }
230 return SendMessageFromPool(broadcast_data, {replica_info});
231 } else {
232 int ret = 0;
233 for (const auto& message : messages) {
234 ret += SendMessageInternal(*message, {replica_info});

240int ReplicaCommunicator::SendMessageFromPool(
241 const google::protobuf::Message& message,
242 const std::vector<ReplicaInfo>& replicas) {
243 int ret = 0;
244 std::string data;
245 message.SerializeToString(&data);
246 global_stats_->SendBroadCastMsgPerRep();
247 std::lock_guard<std::mutex> lk(mutex_);
248 for (const auto& replica : replicas) {
249 auto client = GetClientFromPool(replica.ip(), replica.port());
250 if (client == nullptr) {
251 continue;
252 }
253 //LOG(ERROR) << "send to:" << replica.ip();
254 if (client->SendMessage(data) == 0) {
255 ret++;
256 } else {
257 LOG(ERROR) << "send to:" << replica.ip() << " fail";
258 }
259 //LOG(ERROR) << "send to:" << replica.ip()<<" done";
260 }
261 return ret;
262}

264int ReplicaCommunicator::SendMessageInternal(
265 const google::protobuf::Message& message,
266 const std::vector<ReplicaInfo>& replicas) {
267 int ret = 0;
268 for (const auto& replica : replicas) {
269 auto client = GetClient(replica.ip(), replica.port());
270 if (client == nullptr) {
271 continue;
272 }
273 if (verifier_ != nullptr) {
274 client->SetSignatureVerifier(verifier_);
275 }
276 if (client->SendRawMessage(message) == 0) {
277 ret++;
278 }
279 }
280 return ret;
281}

306void ReplicaCommunicator::SendMessage(const google::protobuf::Message& message,
307 int64_t node_id) {
308 ReplicaInfo target_replica;
309 for (const auto& replica : replicas_) {
310 if (replica.id() == node_id) {
311 target_replica = replica;
312 break;
313 }
314 }
315 if (target_replica.ip().empty()) {
316 for (const auto& replica : GetClientReplicas()) {
317 if (replica.id() == node_id) {
318 target_replica = replica;
319 break;
320 }
321 }
322 }
323
324 if (target_replica.ip().empty()) {
325 LOG(ERROR) << "no replica info";
326 return;
327 }
328
329 int ret = SendMessage(message, target_replica);
330 if (ret < 0) {
331 LOG(ERROR) << "broadcast request fail:";
332 }
333}
