36def serialize(data):
37 """Serialize a dict into a JSON formatted string.
38
39 This function enforces rules like the separator and order of keys.
40 This ensures that all dicts are serialized in the same way.
41
42 This is specially important for hashing data. We need to make sure that
43 everyone serializes their data in the same way so that we do not have
44 hash mismatches for the same structure due to serialization
45 differences.
46
47 Args:
48 data (dict): dict to serialize
49
50 Returns:
51 str: JSON formatted string
52
53 """
54 return rapidjson.dumps(data, skipkeys=False, ensure_ascii=False,
55 sort_keys=True)

107def validate_all_keys_in_obj(obj_name, obj, validation_fun):
108 """Validate all (nested) keys in `obj` by using `validation_fun`.
109
110 Args:
111 obj_name (str): name for `obj` being validated.
112 obj (dict): dictionary object.
113 validation_fun (function): function used to validate the value
114 of `key`.
115
116 Returns:
117 None: indicates validation successful
118
119 Raises:
120 ValidationError: `validation_fun` will raise this error on failure
121 """
122 for key, value in obj.items():
123 validation_fun(obj_name, key)
124 if isinstance(value, dict):
125 validate_all_keys_in_obj(obj_name, value, validation_fun)
126 elif isinstance(value, list):

130def validate_all_values_for_key_in_obj(obj, key, validation_fun):
131 """Validate value for all (nested) occurrence of `key` in `obj`
132 using `validation_fun`.
133
134 Args:
135 obj (dict): dictionary object.
136 key (str): key whose value is to be validated.
137 validation_fun (function): function used to validate the value
138 of `key`.
139
140 Raises:
141 ValidationError: `validation_fun` will raise this error on failure
142 """
143 for vkey, value in obj.items():
144 if vkey == key:
145 validation_fun(value)
146 elif isinstance(value, dict):
147 validate_all_values_for_key_in_obj(value, key, validation_fun)
148 elif isinstance(value, list):

160def validate_key(obj_name, key):
161 """Check if `key` contains ".", "$" or null characters.
162
163 https://docs.mongodb.com/manual/reference/limits/#Restrictions-on-Field-Names
164
165 Args:
166 obj_name (str): object name to use when raising exception
167 key (str): key to validated
168
169 Returns:
170 None: validation successful
171
172 Raises:
173 ValidationError: will raise exception in case of regex match.
174 """
175 if re.search(r'^[$]|\.|\x00', key):
176 error_str = ('Invalid key name "{}" in {} object. The '
177 'key name cannot contain characters '
178 '".", "$" or null characters').format(key, obj_name)
179 raise ValidationError(error_str)
